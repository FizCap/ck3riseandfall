namespace = warlord

# Liege-facing notification: informs the ruler that warlords are rising among vassals
warlord.1001 = {
    type = character_event
    title = warlord.1001.t
    desc = warlord.1001.d
    theme = realm

    option = {
        name = warlord.1001.a
        ai_chance = { factor = 1 }
        # Grant the liege a realm law and run the hidden_effect so the iteration runs in effect scope
        add_realm_law = crown_authority_0
        hidden_effect = {
            every_vassal = {
                limit = {
                    NOT = { has_trait = warlord }
                    NOT = { has_trait = loyal }
                }
                trigger_event = { id = warlord.1002 }
            }
        }
    # Add 30 stability to the liege when this event fires
    change_variable = { name = riseandfall_realm_stability_score add = 30 }
    }
}

# Vassal-facing decision: vassals may choose to become a warlord or remain loyal
warlord.1002 = {
    type = character_event
    title = warlord.1002.t
    desc = warlord.1002.d
    theme = realm

    immediate = {
        # Present a choice; effects happen in options below
    }

    option = {
        name = warlord.1002.a
        # Become a warlord: gain the 'warlord' trait and create a decision to raise faction/claim
        add_trait = warlord
        if = {
            limit = { exists = liege }
            liege = {
                add_opinion = {
                    target = root
                    modifier = warlord_rebellion_opinion
                }
            }
        }
    # Immediately run the warlord action event for this new warlord
    trigger_event = { id = warlord.1003 }
        ai_chance = {
            factor = 10
            modifier = {
                factor = 0
                OR = {
                    exists = liege
                    liege = { opinion = { who = root value >= 50 } }
                    has_trait = craven
                }
            }
        }
    }

    option = {
        name = warlord.1002.b
        # Remain loyal: grant the built-in 'loyal' trait (vanilla trait) or a custom loyal marker
        add_trait = loyal
        # small positive opinion or stability bonus could be added by scripted effects
        ai_chance = {
            factor = 0
            modifier = {
                factor = 90
                OR = {
                    exists = liege
                    liege = { opinion = { who = root value >= 50 } }
                    has_trait = craven
                }
            }
        }
    }
}

# Warlord event: triggered for rulers with the warlord trait
warlord.1003 = {
    type = character_event
    title = warlord.1003.t
    desc = warlord.1003.d
    theme = realm

    option = {
        name = warlord.1003.a
        ai_chance = { factor = 1 }
        # Give claims on the liege's own titles to the warlord
        if = {
            limit = { exists = liege }
            liege = {
                # Give claims on the liege's own county/duchy/kingdom/empire titles to the warlord
                every_held_title = {
                    limit = {
                        tier = tier_county
                        NOT = { root = { has_claim_on = prev } }
                    }
                    root = { add_unpressed_claim = prev }
                }
                every_held_title = {
                    limit = {
                        tier = tier_duchy
                        NOT = { root = { has_claim_on = prev } }
                    }
                    root = { add_unpressed_claim = prev }
                }
                every_held_title = {
                    limit = {
                        tier = tier_kingdom
                        NOT = { root = { has_claim_on = prev } }
                    }
                    root = { add_unpressed_claim = prev }
                }
                every_held_title = {
                    limit = {
                        tier = tier_empire
                        NOT = { root = { has_claim_on = prev } }
                    }
                    root = { add_unpressed_claim = prev }
                }
                if = {
                    limit = { var:riseandfall_realm_stability_score <= 40 }
                    add_legitimacy = { value = legitimacy divide = 98 multiply = -1 }
                }
            }
        }
    }

    option = {
        name = warlord.1003.b
        # Remove the warlord trait and abandon the path
        remove_trait = warlord
        if = {
            limit = { exists = liege }
            liege = {
            }
        }
        ai_chance = {
            factor = 0
            modifier = {
                factor = 100
                exists = liege
                liege = { opinion = { who = root value >= 50 } }
            }
            modifier = {
                factor = 0
                NOT = {
                    exists = liege
                    liege = { opinion = { who = root value >= 50 } }
                }
            }
        }
    }
}
