riseandfall_diadochi_browser.0002 = {
	type = character_event
	title = riseandfall_diadochi_browser.0002.t
	desc = riseandfall_diadochi_browser.0002.desc
	theme = war

	trigger = {
		exists = scope:rf_diadochi_browser_selected_ruler
		exists = scope:rf_diadochi_browser_selected_title
		has_variable = rf_diadochi_browser_self_strength
		has_variable = rf_diadochi_browser_target_strength
	}

	left_portrait = {
		character = root
		animation = personality_bold
	}

	right_portrait = {
		character = scope:rf_diadochi_browser_selected_ruler
		animation = personality_bold
	}

	# Rebuild the list when returning so stale eligibility checks are not reused.
	option = {
		name = riseandfall_diadochi_browser.0002.a
		custom_tooltip = riseandfall_diadochi_browser.0002.a.tt
		hidden_effect = {
			clear_saved_scope = rf_diadochi_browser_selected_ruler
			clear_saved_scope = rf_diadochi_browser_selected_title
			remove_variable = rf_diadochi_browser_self_strength
			remove_variable = rf_diadochi_browser_target_strength
			riseandfall_diadochi_prepare_browser_targets_se = yes
			if = {
				limit = { exists = scope:rf_diadochi_browser_target_ruler_1 }
				trigger_event = { id = riseandfall_diadochi_browser.0001 delayed = yes }
			}
		}
	}

	option = {
		name = riseandfall_diadochi_browser.0002.b
	}

	after = {
		clear_saved_scope = rf_diadochi_browser_selected_ruler
		clear_saved_scope = rf_diadochi_browser_selected_title
		remove_variable = rf_diadochi_browser_self_strength
		remove_variable = rf_diadochi_browser_target_strength
	}
}
