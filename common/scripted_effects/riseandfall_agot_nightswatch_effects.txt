# Rise and Fall - AGOT Night's Watch Compatibility

riseandfall_agot_restore_nights_watch_government_effect = {
    if = {
        limit = {
            is_landed = yes
            has_title = title:k_the_wall
            OR = {
                has_trait = nightswatch
                has_trait = nightswatch_temp
                has_trait = nightswatch_historical
            }
            OR = {
                NOT = { government_has_flag = government_is_nw }
                NOT = { has_realm_law = nights_watch_realm_succession_law }
            }
        }
        if = {
            limit = { NOT = { government_has_flag = government_is_nw } }
            change_government = nights_watch_government
        }
        if = {
            limit = { NOT = { has_realm_law = nights_watch_realm_succession_law } }
            add_realm_law_skip_effects = nights_watch_realm_succession_law
        }
    }
}
