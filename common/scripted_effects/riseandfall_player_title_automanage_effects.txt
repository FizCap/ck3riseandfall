# These mirror the AI automation logic but are gated by character flags

# Player: auto-create kingdom titles when the ruler has the toggle modifier
riseandfall_player_create_kingdoms_se = {
    # Called for a specific ruler (the caller should iterate every_ruler is_ai = no).
    save_scope_as = acting_ruler

    # Only proceed for rulers who are landed (manual create behavior)
    if = {
        limit = { is_landed = yes }

        ordered_alert_creatable_title = {
            order_by = tier
            check_range_bounds = no
            max = 99

            limit = {
                tier = tier_kingdom
                scope:acting_ruler = {
                    OR = {
                        primary_title.tier = tier_empire
                        government_allows = administrative
                        has_realm_law = partition_succession_law
                        has_realm_law = confederate_partition_succession_law
                        has_realm_law = high_partition_succession_law
                    }
                }
                NOT = { tier = scope:acting_ruler.primary_title.tier }
            }

            save_scope_as = target_title

            scope:acting_ruler = {
                create_title_and_vassal_change = { type = created save_scope_as = title_change add_claim_on_loss = no }

                prev = { change_title_holder = { holder = scope:acting_ruler change = scope:title_change } }

                resolve_title_and_vassal_change = scope:title_change

                if = {
                    limit = { primary_title.tier = tier_kingdom }
                    if = {
                        limit = {
                            prev = { holder = { monthly_character_income > scope:acting_ruler.primary_title.holder.monthly_character_income } }
                        }
                        set_primary_title_to = prev
                    }
                }
            }
        }
    }
}

# Player: transfer eligible direct vassals to their rightful de jure liege.
riseandfall_player_transfer_rightful_vassals_se = {
    # CONTRACT
    # Inputs: current ruler scope with direct landed vassals.
    # Outputs: eligible direct vassals are reassigned to their de jure holder or that holder's direct liege.
    # Data shape: character liege changes via create_title_and_vassal_change / change_liege / resolve_title_and_vassal_change.
    # Error modes: missing de jure holders, wars, rank mismatches, or government edge cases skip the transfer.

    save_scope_as = actor

    if = {
        limit = { is_landed = yes }

        every_vassal = {
            limit = {
                liege = scope:actor
                highest_held_title_tier >= tier_county
                exists = primary_title
                exists = primary_title.de_jure_liege
                primary_title.de_jure_liege.holder != liege
                NOT = { is_at_war_with = scope:actor }
                NOT = {
                    any_vassal_or_below = {
                        is_at_war_with = scope:actor
                    }
                }
                NAND = {
                    is_house_head = yes
                    house.house_confederation.leading_house = house
                }
            }

            save_temporary_scope_as = vassal_to_transfer

            if = {
                limit = {
                    exists = primary_title.de_jure_liege.holder
                    primary_title.de_jure_liege.holder = {
                        liege = scope:actor
                        highest_held_title_tier > scope:vassal_to_transfer.highest_held_title_tier
                        NOR = {
                            is_at_war_with = scope:actor
                            is_at_war_with = scope:vassal_to_transfer
                        }
                    }
                    NAND = {
                        primary_title.de_jure_liege.holder.liege ?= { has_government = japan_administrative_government }
                        primary_title.de_jure_liege.holder ?= { has_government = japan_administrative_government }
                        has_government = japan_feudal_government
                    }
                }

                primary_title.de_jure_liege.holder = { save_temporary_scope_as = transfer_recipient }

                scope:actor = {
                    create_title_and_vassal_change = {
                        type = granted
                        save_scope_as = change
                        add_claim_on_loss = no
                    }
                }

                change_liege = {
                    liege = scope:transfer_recipient
                    change = scope:change
                }

                scope:actor = { resolve_title_and_vassal_change = scope:change }

                scope:transfer_recipient = {
                    add_opinion = {
                        target = scope:actor
                        modifier = granted_vassal
                    }
                }
            }
            else_if = {
                limit = {
                    exists = primary_title.de_jure_liege.holder.liege
                    primary_title.de_jure_liege.holder.liege = {
                        this != scope:actor
                        liege = scope:actor
                        highest_held_title_tier > scope:vassal_to_transfer.highest_held_title_tier
                        NOR = {
                            is_at_war_with = scope:actor
                            is_at_war_with = scope:vassal_to_transfer
                        }
                    }
                    NAND = {
                        primary_title.de_jure_liege.holder.liege ?= { has_government = japan_administrative_government }
                        primary_title.de_jure_liege.holder ?= { has_government = japan_administrative_government }
                        has_government = japan_feudal_government
                    }
                }

                primary_title.de_jure_liege.holder.liege = { save_temporary_scope_as = transfer_recipient }

                scope:actor = {
                    create_title_and_vassal_change = {
                        type = granted
                        save_scope_as = change
                        add_claim_on_loss = no
                    }
                }

                change_liege = {
                    liege = scope:transfer_recipient
                    change = scope:change
                }

                scope:actor = { resolve_title_and_vassal_change = scope:change }

                scope:transfer_recipient = {
                    add_opinion = {
                        target = scope:actor
                        modifier = granted_vassal
                    }
                }
            }
        }
    }
}

# Player: auto-create duchies when the ruler/character has the toggle modifier
riseandfall_player_create_duchies_se = {
    # Called for a specific ruler (the caller should iterate every_ruler is_ai = no).
    save_scope_as = acting_count

    if = {
        limit = { is_landed = yes }

        ordered_alert_creatable_title = {
            order_by = tier
            check_range_bounds = no
            max = 99

            limit = { tier = tier_duchy }

            save_scope_as = target_title

            scope:acting_count = {
                create_title_and_vassal_change = { type = created save_scope_as = title_change add_claim_on_loss = no }

                prev = { change_title_holder = { holder = scope:acting_count change = scope:title_change } }

                resolve_title_and_vassal_change = scope:title_change

                if = { limit = { primary_title.tier = tier_county } set_primary_title_to = prev }
            }
        }
    }
}

# Warlord: create every uncreated de jure duchy, kingdom, and empire where the
# warlord holds 51%+ of de jure counties (vanilla requirement enforced via
# alert_creatable_title). Duchies/kingdoms/empires are created in ascending
# tier order so duchy titles exist before kingdom titles are claimed.
riseandfall_warlord_create_realm_titles_se = {
    save_scope_as = acting_warlord

    if = {
        limit = {
            is_landed = yes
            has_trait = warlord
        }

        # Duchies first (tier_duchy), then kingdoms, then empires.
        # every_alert_creatable_title uses the same engine check as the
        # vanilla title-creation UI: the warlord (or their realm) must hold
        # at least 51% of the title's de jure counties.
        every_alert_creatable_title = {
            limit = { tier = tier_duchy }
            save_scope_as = target_title

            scope:acting_warlord = {
                create_title_and_vassal_change = { type = created save_scope_as = title_change add_claim_on_loss = no }
                scope:target_title = { change_title_holder = { holder = scope:acting_warlord change = scope:title_change } }
                resolve_title_and_vassal_change = scope:title_change
                if = { limit = { primary_title.tier = tier_county } set_primary_title_to = scope:target_title }
            }
        }

        every_alert_creatable_title = {
            limit = { tier = tier_kingdom }
            save_scope_as = target_title

            scope:acting_warlord = {
                create_title_and_vassal_change = { type = created save_scope_as = title_change add_claim_on_loss = no }
                scope:target_title = { change_title_holder = { holder = scope:acting_warlord change = scope:title_change } }
                resolve_title_and_vassal_change = scope:title_change
                if = { limit = { scope:target_title.tier > primary_title.tier } set_primary_title_to = scope:target_title }
            }
        }

        every_alert_creatable_title = {
            limit = { tier = tier_empire }
            save_scope_as = target_title

            scope:acting_warlord = {
                create_title_and_vassal_change = { type = created save_scope_as = title_change add_claim_on_loss = no }
                scope:target_title = { change_title_holder = { holder = scope:acting_warlord change = scope:title_change } }
                resolve_title_and_vassal_change = scope:title_change
                if = { limit = { scope:target_title.tier > primary_title.tier } set_primary_title_to = scope:target_title }
            }
        }
    }
}

# Player: auto-grant duchy titles to de jure vassals (uses AI handout logic adapted)
riseandfall_player_grant_duchies_se = {
    # Expects to be called in a ruler scope (the caller should iterate every_ruler is_ai = no).
    save_scope_as = actor

    if = { limit = { is_landed = yes }

        # Iterate de-jure duchies in the realm
        every_realm_de_jure_duchy = {
            limit = { exists = this holder = scope:actor }
            save_scope_as = target_title

            if = {
                limit = { scope:target_title = { holder = scope:actor } }

                # Don't hand out duchies that contain the actor's capital
                if = {
                    limit = { NOT = { scope:target_title = { any_de_jure_county = { this = scope:actor.capital_province.county } } } }

                    scope:target_title = {
                                        random_dejure_vassal_title_holder = {
                                            limit = { any_liege_or_above = { this = scope:actor } NOT = { this = scope:actor } highest_held_title_tier >= tier_county }
                                            save_scope_as = chosen_vassal
                                        }
                    }

                    if = {
                        limit = {
                            exists = scope:chosen_vassal
                            scope:chosen_vassal = { any_liege_or_above = { this = scope:actor } }
                            OR = {
                                NOT = { scope:chosen_vassal = { government_has_flag = government_is_theocracy } }
                                scope:actor = { government_has_flag = government_is_theocracy }
                            }
                        }

                        # Don't grant titles that have giveaway blocked when the actor is a human player.
                        if = {
                            limit = { NOT = { AND = { scope:target_title = { has_variable = Riseandfall_giveaway_blocked } scope:actor = { is_ai = no } } } }
                            scope:chosen_vassal = {
                                hidden_effect = {
                                    add_opinion = {
                                        target = scope:actor
                                        modifier = received_title_duchy
                                    }
                                }
                            }
                            create_title_and_vassal_change = { type = granted save_scope_as = change }
                            scope:target_title = { change_title_holder_include_vassals = { holder = scope:chosen_vassal change = scope:change take_baronies = no } }
                            resolve_title_and_vassal_change = scope:change
                        }
                    }
                    else = {
                        if = {
                            limit = {
                                exists = scope:chosen_vassal
                                scope:chosen_vassal = { government_has_flag = government_is_theocracy }
                                NOT = { scope:actor = { government_has_flag = government_is_theocracy } }
                            }
                        }
                    }
                }
            }
        }
    }
}

# Player: auto-grant kingdoms when the ruler has the toggle (same adaptation pattern as AI)
riseandfall_player_grant_kingdoms_se = {
    # Expects to be called in a ruler scope (the caller should iterate every_ruler is_ai = no).
    save_scope_as = actor

    if = { limit = { is_landed = yes }

        every_realm_de_jure_kingdom = {
            limit = { exists = this holder = scope:actor }
            save_scope_as = target_title

            if = {
                limit = { scope:target_title = { holder = scope:actor } }

                if = {
                    limit = {
                        NOT = { scope:actor = { primary_title = { de_jure_liege = scope:target_title } } }
                        NOT = { scope:target_title = { any_de_jure_county = { this = scope:actor.capital_province.county } } }
                    }

                    scope:target_title = {
                        # Prefer dukes first, then counts, then kings for player auto-granting of kingdoms.
                        random_dejure_vassal_title_holder = {
                            limit = { is_landed = yes any_liege_or_above = { this = scope:actor } NOT = { this = scope:actor } highest_held_title_tier >= tier_duchy }
                            save_scope_as = chosen_vassal
                        }

                        if = {
                            limit = { NOT = { exists = scope:chosen_vassal } }
                            random_dejure_vassal_title_holder = {
                                limit = { is_landed = yes any_liege_or_above = { this = scope:actor } NOT = { this = scope:actor } highest_held_title_tier >= tier_county }
                                save_scope_as = chosen_vassal
                            }
                        }

                        if = {
                            limit = { NOT = { exists = scope:chosen_vassal } }
                            random_dejure_vassal_title_holder = {
                                limit = { is_landed = yes any_liege_or_above = { this = scope:actor } NOT = { this = scope:actor } highest_held_title_tier >= tier_kingdom }
                                save_scope_as = chosen_vassal
                            }
                        }
                    }

                    if = {
                        limit = {
                            exists = scope:chosen_vassal
                            scope:chosen_vassal = { any_liege_or_above = { this = scope:actor } }
                            OR = {
                                NOT = { scope:chosen_vassal = { government_has_flag = government_is_theocracy } }
                                scope:actor = { government_has_flag = government_is_theocracy }
                            }
                        }

                        # Skip granting if the target title has the giveaway-block variable AND the actor is a human player.
                        if = {
                            limit = { NOT = { AND = { scope:target_title = { has_variable = Riseandfall_giveaway_blocked } scope:actor = { is_ai = no } } } }
                            scope:chosen_vassal = {
                                hidden_effect = {
                                    add_opinion = {
                                        target = scope:actor
                                        modifier = received_title_kingdom
                                    }
                                }
                            }
                            create_title_and_vassal_change = { type = granted save_scope_as = change }
                            scope:target_title = { change_title_holder_include_vassals = { holder = scope:chosen_vassal change = scope:change take_baronies = no } }
                            resolve_title_and_vassal_change = scope:change

                            # Transfer de jure duchies under the granted kingdom to the new holder
                            scope:target_title = {
                                    every_direct_de_jure_vassal_title = {
                                        limit = { tier = tier_duchy holder != scope:chosen_vassal }
                                        create_title_and_vassal_change = { type = granted save_scope_as = duchy_change }
                                        change_title_holder_include_vassals = { holder = scope:chosen_vassal change = scope:duchy_change take_baronies = no }
                                        resolve_title_and_vassal_change = scope:duchy_change
                                    }
                            }
                        }
                    }
                }
            }
        }
    }
}
