# Keep the Puppet Master court position aligned to the active puppet-regency diarch.
# Expected scope: root = liege
riseandfall_sync_puppet_master_court_position = {
	clear_saved_scope = rf_puppet_master_liege
	clear_saved_scope = rf_current_puppet_master_diarch
	clear_saved_scope = rf_current_puppet_master_holder
	save_scope_as = rf_puppet_master_liege

	if = {
		limit = {
			has_diarchy_type = puppet_regency
			has_active_diarchy = yes
			exists = diarch
			diarch = {
				is_alive = yes
			}
		}
		diarch = { save_scope_as = rf_current_puppet_master_diarch }
	}

	if = {
		limit = { employs_court_position = riseandfall_puppet_master_court_position }
		random_court_position_holder = {
			type = riseandfall_puppet_master_court_position
			save_scope_as = rf_current_puppet_master_holder
		}
	}

	if = {
		limit = {
			exists = scope:rf_current_puppet_master_holder
			scope:rf_current_puppet_master_holder = {
				has_court_position = riseandfall_puppet_master_court_position
			}
			NOT = { exists = scope:rf_current_puppet_master_diarch }
		}
		scope:rf_puppet_master_liege = {
			revoke_court_position = {
				recipient = scope:rf_current_puppet_master_holder
				court_position = riseandfall_puppet_master_court_position
			}
		}
	}

	if = {
		limit = {
			exists = scope:rf_puppet_master_liege
			exists = scope:rf_current_puppet_master_diarch
			scope:rf_current_puppet_master_diarch = {
				is_alive = yes
			}
		}
		scope:rf_current_puppet_master_diarch = {
			if = {
				limit = {
					is_ruler = no
					NOT = { employer = scope:rf_puppet_master_liege }
				}
				set_employer = scope:rf_puppet_master_liege
			}
		}
		if = {
			limit = {
				exists = scope:rf_current_puppet_master_holder
				scope:rf_current_puppet_master_holder = {
					has_court_position = riseandfall_puppet_master_court_position
				}
				scope:rf_current_puppet_master_diarch = {
					is_ruler = no
					employer = scope:rf_puppet_master_liege
				}
				NOT = { scope:rf_current_puppet_master_holder ?= scope:rf_current_puppet_master_diarch }
			}
			scope:rf_puppet_master_liege = {
				replace_court_position = {
					recipient = scope:rf_current_puppet_master_diarch
					holder = scope:rf_current_puppet_master_holder
					court_position = riseandfall_puppet_master_court_position
				}
			}
		}
		else_if = {
			limit = {
				NOT = { exists = scope:rf_current_puppet_master_holder }
				scope:rf_current_puppet_master_diarch = {
					is_ruler = no
					NOT = { has_court_position = riseandfall_puppet_master_court_position }
					employer = scope:rf_puppet_master_liege
				}
			}
			scope:rf_puppet_master_liege = {
				appoint_court_position = {
					recipient = scope:rf_current_puppet_master_diarch
					court_position = riseandfall_puppet_master_court_position
				}
			}
		}
	}
}

# Claim the puppet master office when the active diarch is not currently the court position holder
riseandfall_claim_puppet_master_office_effect = {
	scope:recipient = {
		if = {
			limit = {
				scope:actor = {
					is_alive = yes
					NOT = { is_ruler = yes }
				}
			}
			if = {
				limit = {
					NOR = {
						scope:actor = { is_courtier_of = scope:recipient }
						scope:actor = { is_vassal_of = scope:recipient }
						scope:actor = { employer = scope:recipient }
					}
				}
				scope:actor = { set_employer = scope:recipient }
			}

			if = {
				limit = {
					employs_court_position = riseandfall_puppet_master_court_position
					scope:actor = {
						OR = {
							is_courtier_of = scope:recipient
							employer = scope:recipient
						}
					}
				}
				random_court_position_holder = {
					type = riseandfall_puppet_master_court_position
					save_scope_as = old_puppet_master
				}
				replace_court_position = {
					recipient = scope:actor
					holder = scope:old_puppet_master
					court_position = riseandfall_puppet_master_court_position
				}
			}
			else = {
				if = {
					limit = {
						scope:actor = {
							OR = {
								is_courtier_of = scope:recipient
								employer = scope:recipient
							}
						}
					}
					appoint_court_position = {
						recipient = scope:actor
						court_position = riseandfall_puppet_master_court_position
					}
				}
			}
		}
	}
}

# Steal 5% of liege short-term gold and give it to diarch
riseandfall_steal_gold_effect = {
	# actor = diarch, recipient = liege
	# Based on palace coup gold transfer
	scope:recipient = {
		# Initialize the transfer amount so change_variable always operates on a value variable
		set_variable = { name = rf_transfer_gold value = 0 }
		# If the liege has gold, set the transfer amount to their gold first
		if = { limit = { gold > 1 } set_variable = { name = rf_transfer_gold value = gold } }
		# Use integer math: divide by 20 to transfer roughly 5%
		if = { limit = { var:rf_transfer_gold > 0 } change_variable = { name = rf_transfer_gold divide = 20 } }
	}
	# Give to diarch
	if = { limit = { scope:recipient = { has_variable = rf_transfer_gold } scope:recipient = { var:rf_transfer_gold > 0 } } scope:actor = { add_short_term_gold = scope:recipient.var:rf_transfer_gold } }
	# Remove from liege
	scope:recipient = {
		if = { limit = { var:rf_transfer_gold > 0 } remove_short_term_gold = var:rf_transfer_gold }
		remove_variable = rf_transfer_gold
	}

	# Using this interaction spends 10 points of puppet stability mana.
	scope:recipient = {
		if = {
			limit = { has_variable = riseandfall_rs_puppet_stability_bonus }
			change_variable = { name = riseandfall_rs_puppet_stability_bonus add = -10 }
			clamp_variable = { name = riseandfall_rs_puppet_stability_bonus min = -100 max = 100 }
			round_variable = { name = riseandfall_rs_puppet_stability_bonus nearest = 1 }
		}
	}
}

# When a guest arrives at a court, ensure a visiting puppet master (diarch)
# who is a foreign guest (not already a courtier) becomes employed by the host
# so they are treated as a courtier in that court.
# Expected invocation context: root = guest, host = host
riseandfall_set_guest_puppet_master_employer_se = {
	# Run in the guest scope (root)
	root = {
		if = {
			limit = {
				# Is the root character the active diarch of the host?
				is_diarch_of_target = scope:host
				# Is the root a foreign court guest of the host (they have a liege)
				is_foreign_court_guest_of = scope:host
				# Rulers cannot be moved into another ruler's court.
				is_ruler = no
				# Make sure they are not already a courtier of the host
				NOT = { is_courtier_of = scope:host }
			}
			# Move the guest into the host's court (set them as employed by the host)
			set_employer = scope:host
		}
	}
}

	riseandfall_overthrow_liege_effect = {
	# Based on palace coup overthrow
	create_title_and_vassal_change = {
		type = conquest
		save_scope_as = change
	}
	# Transfer all held titles from recipient to actor
	scope:recipient = {
		if = {
			limit = { has_active_diarchy = yes }
			end_diarchy = yes
		}
		riseandfall_sync_puppet_master_court_position = yes
        # If the recipient has a capital county, transfer that county first
        if = {
            limit = { scope:recipient = { capital_county = { exists = yes } } }
            scope:recipient = {
                capital_county = {
                    change_title_holder = {
                        holder = scope:actor
                        change = scope:change
                        take_baronies = yes
                    }
                }
            }
        }
		# Now iterate through all held titles explicitly and assign them to actor
		scope:recipient = {
			every_held_title = {
				limit = {
					# Don't transfer the baronies again if we transfer the top county, but we just skip baronies
					NOT = { tier = tier_barony }
					NOT = { is_noble_family_title = yes }
					NOT = { is_head_of_faith = yes }
				}
				change_title_holder = {
					holder = scope:actor
					change = scope:change
					take_baronies = yes
				}
			}
		}
		every_vassal = {
			limit = {
				NOT = { this = scope:actor }
				liege = scope:recipient
			}
			change_liege = {
				liege = scope:actor
				change = scope:change
			}
		}
		if = {
			limit = { is_landed = yes }
			change_liege = {
				liege = scope:actor
				change = scope:change
			}
		}
		else = {
			set_employer = scope:actor
		}
	}
	resolve_title_and_vassal_change = scope:change

}

# Force liege to abdicate by transferring their titles to a chosen successor (youngest unlanded child; fallback to dynasty in court)
riseandfall_force_abdicate_effect = {
	scope:recipient = {
		# Using this interaction also spends 10 points of puppet stability mana.
		if = {
			limit = { has_variable = riseandfall_rs_puppet_stability_bonus }
			change_variable = { name = riseandfall_rs_puppet_stability_bonus add = -10 }
			clamp_variable = { name = riseandfall_rs_puppet_stability_bonus min = -100 max = 100 }
			round_variable = { name = riseandfall_rs_puppet_stability_bonus nearest = 1 }
		}

		# initialize a big age to find the youngest
		set_variable = { name = rf_candidate_best_age value = 999 }
		# Choose candidate depending on realm succession law: prefer males/females per law settings
		if = {
			limit = { scope:recipient = { has_realm_law = male_only_law } }
			# Male-only: pick youngest unlanded male child
			every_close_or_extended_family_member = {
				limit = {
					is_landed = no
					NOT = { is_adult = yes }
					is_male = yes
					OR = {
						employer = scope:recipient
						liege = scope:recipient
						top_liege = scope:recipient
					}
				}
				set_variable = { name = candidate_age value = age }
				save_scope_as = candidate_for_score
				scope:recipient = { set_variable = { name = tmp_candidate_age value = scope:candidate_for_score.var:candidate_age } }
				if = {
					limit = { scope:recipient = { has_variable = tmp_candidate_age has_variable = rf_candidate_best_age var:tmp_candidate_age < var:rf_candidate_best_age } }
					save_scope_as = chosen_successor
					scope:recipient = { set_variable = { name = rf_candidate_best_age value = scope:recipient.var:tmp_candidate_age } }
				}
			}
		}
		else_if = {
			limit = { scope:recipient = { has_realm_law = male_preference_law } }
			# Male-preference: try youngest males first, then females if none found
			# First pass: males
			every_close_or_extended_family_member = {

					is_landed = no
					NOT = { is_adult = yes }
					is_male = yes
					OR = {
						employer = scope:recipient
						liege = scope:recipient
						top_liege = scope:recipient

				}
			}
			# If no male chosen, try females
			if = {
				limit = { NOT = { exists = scope:chosen_successor } }
				scope:recipient = { set_variable = { name = rf_candidate_best_age value = 999 } }
				every_close_or_extended_family_member = {
					limit = {
						is_landed = no
						NOT = { is_adult = yes }
						is_male = no
						OR = {
							employer = scope:recipient

						save_scope_as = chosen_successor

					save_scope_as = chosen_successor

			if = {
				limit = { NOT = { exists = scope:chosen_successor } }

			every_close_or_extended_family_member = {

			limit = { NOT = { exists = scope:chosen_successor } scope:recipient = { has_dynasty = yes } }

				limit = {
					exists = scope:chosen_successor

					# Puppet-master diarch must be a valid courtier-type character (not a ruler)

				}
				# If no active diarchy exists yet on the chosen_successor, try to start one
				if = {
					limit = { NOT = { scope:chosen_successor = { has_active_diarchy = yes } } }
					scope:actor = {
						if = { limit = { NOT = { employer = scope:chosen_successor } } set_employer = scope:chosen_successor }
					}
					scope:chosen_successor = {
						try_start_diarchy = puppet_regency
						set_diarch = scope:actor
						set_diarchy_type = puppet_regency
						set_diarchy_swing = 90
						riseandfall_sync_puppet_master_court_position = yes
					}
				}
				# Only set diarch/diarchy fields if an active diarchy now exists
				else_if = {
					limit = { scope:chosen_successor = { has_active_diarchy = yes } }
					# Ensure the diarch is in the new liege's court before setting them as diarch
					scope:actor = {
						if = { limit = { NOT = { employer = scope:chosen_successor } } set_employer = scope:chosen_successor }
					}
					scope:chosen_successor = {
						set_diarch = scope:actor
						set_diarchy_type = puppet_regency
						set_diarchy_swing = 90
						riseandfall_sync_puppet_master_court_position = yes
					}
				}
			}
		}
		# Keep court-position assignment consistent if this effect exits early.
		scope:recipient = { riseandfall_sync_puppet_master_court_position = yes }
	}
}
