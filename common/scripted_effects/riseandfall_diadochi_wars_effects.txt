# Rise and Fall: scripted effect to force Diadochi wars among same-origin realms
# Moved from on_action to keep on_action concise.
#
# Design notes:
#  - Iterates every_independent_ruler to find eligible attackers.
#  - For each attacker, picks ONE random peer successor sharing the same
#    rf_successor_origin_title on their primary title.
#  - the war busy check was removed to allow chaotic pile-ons. Attacker and defender
#    are no longer required to be at peace, enabling instant war cascades.
#  - Attackers only strike if their military power is >= 1.2x the target's
#    (or >= 1.0x if the target is already at war).
riseandfall_diadochi_wars_se = {
    every_independent_ruler = {
        limit = {
            # Must be a successor state ruler (primary title carries rf_is_successor_split_title
            # and rf_successor_origin_title, set by riseandfall_execute_realm_split_se).
            riseandfall_has_successor_primary_title_trigger = yes
            is_at_war = no
            is_in_debt = no
            is_ai = yes
        }
        save_scope_as = rf_diadochi_attacker

        # Find one peer successor that shares the same split origin and is a legal war target.
        # Prefer neighbors first (more plausible Diadochi conflicts). Evaluate all valid targets
        # using every_independent_ruler and pick one from the list.
        every_independent_ruler = {
            limit = {
                # Never attack yourself.
                NOT = { this = scope:rf_diadochi_attacker }

                # Valid targets: a peer successor from the same origin, OR the origin ruler themselves.
                # Guard the attacker variable before any title comparison to avoid null matches.
                scope:rf_diadochi_attacker.primary_title = { has_variable = rf_successor_origin_title }
                OR = {
                    # Peer successor state sharing the same split origin.
                    AND = {
                        primary_title = {
                            has_variable = rf_is_successor_split_title
                            has_variable = rf_successor_origin_title
                        }
                        primary_title.var:rf_successor_origin_title = scope:rf_diadochi_attacker.primary_title.var:rf_successor_origin_title
                    }
                    # The origin country ruler (holds the title stored in rf_successor_origin_title).
                    has_title = scope:rf_diadochi_attacker.primary_title.var:rf_successor_origin_title
                }

                # Military Check: Attacker must be 1.2x stronger, or 1.0x if defender is already fighting.
                trigger_if = {
                    limit = { is_at_war = yes }
                    scope:rf_diadochi_attacker.military_power >= military_power
                }
                trigger_else = {
                    scope:rf_diadochi_attacker.military_power >= {
                        value = military_power
                        multiply = 1.2
                    }
                }

                # Neighbor requirement: check from the defender's perspective (borderhood is symmetric).
                # This avoids the prev-scope ambiguity of nesting inside scope:rf_diadochi_attacker.
                any_neighboring_top_liege_realm_owner = {
                    this = scope:rf_diadochi_attacker
                }

                # Explicit truce exclusion.
                NOT = {
                    scope:rf_diadochi_attacker = { has_truce = prev }
                }

                # Full engine-level war eligibility: handles CB legality, etc.
                scope:rf_diadochi_attacker = {
                    can_declare_war = {
                        defender = prev
                        casus_belli = riseandfall_diadochi_cb
                        target_titles = { prev.primary_title }
                    }
                }
            }
            add_to_list = rf_diadochi_targets
        }

        random_in_list = {
            list = rf_diadochi_targets
            save_scope_as = rf_diadochi_defender
        }

        # Fallback: if no neighbor was found, allow same-origin non-neighbors.
        if = {
            limit = { NOT = { exists = scope:rf_diadochi_defender } }
            every_independent_ruler = {
                limit = {
                    NOT = { this = scope:rf_diadochi_attacker }

                    # Valid targets: a peer successor from the same origin, OR the origin ruler themselves.
                    # Guard the attacker variable before any title comparison to avoid null matches.
                    scope:rf_diadochi_attacker.primary_title = { has_variable = rf_successor_origin_title }
                    OR = {
                        AND = {
                            primary_title = {
                                has_variable = rf_is_successor_split_title
                                has_variable = rf_successor_origin_title
                            }
                            primary_title.var:rf_successor_origin_title = scope:rf_diadochi_attacker.primary_title.var:rf_successor_origin_title
                        }
                        has_title = scope:rf_diadochi_attacker.primary_title.var:rf_successor_origin_title
                    }

                    trigger_if = {
                        limit = { is_at_war = yes }
                        scope:rf_diadochi_attacker.military_power >= military_power
                    }
                    trigger_else = {
                        scope:rf_diadochi_attacker.military_power >= {
                            value = military_power
                            multiply = 1.2
                        }
                    }

                    NOT = {
                        scope:rf_diadochi_attacker = { has_truce = prev }
                    }
                    
                    scope:rf_diadochi_attacker = {
                        can_declare_war = {
                            defender = prev
                            casus_belli = riseandfall_diadochi_cb
                            target_titles = { prev.primary_title }
                        }
                    }
                }
                add_to_list = rf_diadochi_fallback_targets
            }
            random_in_list = {
                list = rf_diadochi_fallback_targets
                save_scope_as = rf_diadochi_defender
            }
        }

        # Only proceed if a valid defender was found.
        if = {
            limit = { exists = scope:rf_diadochi_defender }

            # Declare the Diadochi war.
            scope:rf_diadochi_attacker = {
                start_war = {
                    casus_belli = riseandfall_diadochi_cb
                    target = scope:rf_diadochi_defender
                    target_title = scope:rf_diadochi_defender.primary_title
                }
            }
        }
    }
}
