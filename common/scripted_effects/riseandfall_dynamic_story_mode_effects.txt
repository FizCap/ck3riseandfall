# Rise and Fall - Dynamic Story Mode
# Campaign director for world climate, story roles, collapse pressure, and successor grace.

riseandfall_story_world_climate_se = {
    # Yearly campaign read: this is the director's "table state" before individual
    # rulers are scored. Only independent kingdom+ realms are tracked as major
    # story actors; smaller realms still calculate stability, but they should not
    # decide whether the whole world needs more danger or more recovery.
    set_global_variable = { name = riseandfall_story_world_eligible_count value = 0 }
    set_global_variable = { name = riseandfall_story_world_stable_count value = 0 }
    set_global_variable = { name = riseandfall_story_world_unstable_count value = 0 }
    set_global_variable = { name = riseandfall_story_world_crisis_count value = 0 }
    set_global_variable = { name = riseandfall_story_world_empire_count value = 0 }
    set_global_variable = { name = riseandfall_story_world_hegemony_count value = 0 }
    set_global_variable = { name = riseandfall_story_world_blob_count value = 0 }
    set_global_variable = { name = riseandfall_story_world_stable_pct value = 0 }
    set_global_variable = { name = riseandfall_story_world_unstable_pct value = 0 }
    set_global_variable = { name = riseandfall_story_world_crisis_pct value = 0 }
    set_global_variable = { name = riseandfall_story_world_pressure value = 0 }
    set_global_variable = { name = riseandfall_story_world_climate value = 0 }
    set_global_variable = { name = riseandfall_world_total value = 0 }
    set_global_variable = { name = riseandfall_world_total_development value = 0 }

    every_independent_ruler = {
        limit = { is_landed = yes }
        change_global_variable = { name = riseandfall_world_total add = realm_size }
        save_scope_as = rf_story_development_ruler
        set_variable = { name = riseandfall_rs_total_development value = 0 }
        every_realm_county = {
            scope:rf_story_development_ruler = {
                change_variable = { name = riseandfall_rs_total_development add = prev.development_level }
            }
        }
        change_global_variable = { name = riseandfall_world_total_development add = var:riseandfall_rs_total_development }
    }
    if = { limit = { global_var:riseandfall_world_total < 1 } set_global_variable = { name = riseandfall_world_total value = 1 } }
    if = { limit = { global_var:riseandfall_world_total_development < 1 } set_global_variable = { name = riseandfall_world_total_development value = 1 } }

    every_independent_ruler = {
        limit = {
            is_landed = yes
            primary_title.tier >= tier_kingdom
        }
        change_global_variable = { name = riseandfall_story_world_eligible_count add = 1 }

        # Stability bands are tier-relative, matching the low-stability event
        # thresholds used elsewhere: kingdom 40, empire 45, hegemony 50. Larger
        # realms need more cohesion to count as truly safe, and can be in trouble
        # before a flat kingdom threshold would notice.
        set_variable = { name = tmp_story_low_threshold value = 40 }
        if = { limit = { primary_title.tier >= tier_empire } set_variable = { name = tmp_story_low_threshold value = 45 } }
        if = { limit = { primary_title.tier >= tier_hegemony } set_variable = { name = tmp_story_low_threshold value = 50 } }
        set_variable = { name = tmp_story_crisis_threshold value = var:tmp_story_low_threshold }
        change_variable = { name = tmp_story_crisis_threshold add = -15 }
        set_variable = { name = tmp_story_stable_threshold value = var:tmp_story_low_threshold }
        change_variable = { name = tmp_story_stable_threshold add = 30 }

        if = {
            limit = {
                has_variable = riseandfall_realm_stability_score
                var:riseandfall_realm_stability_score >= var:tmp_story_stable_threshold
            }
            change_global_variable = { name = riseandfall_story_world_stable_count add = 1 }
        }
        if = {
            limit = {
                has_variable = riseandfall_realm_stability_score
                var:riseandfall_realm_stability_score < var:tmp_story_low_threshold
            }
            change_global_variable = { name = riseandfall_story_world_unstable_count add = 1 }
        }
        if = {
            limit = {
                has_variable = riseandfall_realm_stability_score
                var:riseandfall_realm_stability_score < var:tmp_story_crisis_threshold
            }
            change_global_variable = { name = riseandfall_story_world_crisis_count add = 1 }
        }
        if = {
            limit = { primary_title.tier >= tier_empire }
            change_global_variable = { name = riseandfall_story_world_empire_count add = 1 }
        }
        if = {
            limit = { primary_title.tier >= tier_hegemony }
            change_global_variable = { name = riseandfall_story_world_hegemony_count add = 1 }
        }

        set_variable = { name = tmp_story_world_total value = global_var:riseandfall_world_total }
        if = { limit = { var:tmp_story_world_total < 1 } set_variable = { name = tmp_story_world_total value = 1 } }
        set_variable = { name = tmp_story_world_development_total value = global_var:riseandfall_world_total_development }
        if = { limit = { var:tmp_story_world_development_total < 1 } set_variable = { name = tmp_story_world_development_total value = 1 } }
        set_variable = { name = tmp_story_realm_pct value = realm_size }
        change_variable = { name = tmp_story_realm_pct multiply = 100 }
        change_variable = { name = tmp_story_realm_pct divide = var:tmp_story_world_total }
        round_variable = { name = tmp_story_realm_pct nearest = 1 }
        set_variable = { name = tmp_story_development_pct value = var:riseandfall_rs_total_development }
        change_variable = { name = tmp_story_development_pct multiply = 100 }
        change_variable = { name = tmp_story_development_pct divide = var:tmp_story_world_development_total }
        round_variable = { name = tmp_story_development_pct nearest = 1 }
        if = {
            limit = {
                OR = {
                    var:tmp_story_realm_pct >= 8
                    var:tmp_story_development_pct >= 8
                    primary_title.tier >= tier_hegemony
                }
            }
            change_global_variable = { name = riseandfall_story_world_blob_count add = 1 }
        }
        remove_variable = tmp_story_world_total
        remove_variable = tmp_story_world_development_total
        remove_variable = tmp_story_realm_pct
        remove_variable = tmp_story_development_pct
        remove_variable = tmp_story_low_threshold
        remove_variable = tmp_story_crisis_threshold
        remove_variable = tmp_story_stable_threshold
    }

    # Convert counts into percentages. These globals are intentionally exposed as
    # coarse campaign signals so individual realms can react to the wider age.
    if = {
        limit = { global_var:riseandfall_story_world_eligible_count > 0 }
        set_global_variable = { name = riseandfall_story_world_stable_pct value = global_var:riseandfall_story_world_stable_count }
        change_global_variable = { name = riseandfall_story_world_stable_pct multiply = 100 }
        change_global_variable = { name = riseandfall_story_world_stable_pct divide = global_var:riseandfall_story_world_eligible_count }
        round_global_variable = { name = riseandfall_story_world_stable_pct nearest = 1 }

        set_global_variable = { name = riseandfall_story_world_unstable_pct value = global_var:riseandfall_story_world_unstable_count }
        change_global_variable = { name = riseandfall_story_world_unstable_pct multiply = 100 }
        change_global_variable = { name = riseandfall_story_world_unstable_pct divide = global_var:riseandfall_story_world_eligible_count }
        round_global_variable = { name = riseandfall_story_world_unstable_pct nearest = 1 }

        set_global_variable = { name = riseandfall_story_world_crisis_pct value = global_var:riseandfall_story_world_crisis_count }
        change_global_variable = { name = riseandfall_story_world_crisis_pct multiply = 100 }
        change_global_variable = { name = riseandfall_story_world_crisis_pct divide = global_var:riseandfall_story_world_eligible_count }
        round_global_variable = { name = riseandfall_story_world_crisis_pct nearest = 1 }
    }

    # World pressure nudges the director: high stability asks for new stress on
    # complacent powers; widespread crisis asks for recovery so the map does not
    # become a uniform death spiral.
    if = {
        limit = {
            global_var:riseandfall_story_world_stable_pct >= 70
            NOT = { global_var:riseandfall_story_world_crisis_pct >= 10 }
        }
        set_global_variable = { name = riseandfall_story_world_pressure value = 2 }
        set_global_variable = { name = riseandfall_story_world_climate value = 1 }
    }
    else_if = {
        limit = { global_var:riseandfall_story_world_stable_pct >= 55 }
        set_global_variable = { name = riseandfall_story_world_pressure value = 1 }
        set_global_variable = { name = riseandfall_story_world_climate value = 1 }
    }
    else_if = {
        limit = { global_var:riseandfall_story_world_crisis_pct >= 25 }
        set_global_variable = { name = riseandfall_story_world_pressure value = -2 }
        set_global_variable = { name = riseandfall_story_world_climate value = 4 }
    }
    else_if = {
        limit = { global_var:riseandfall_story_world_unstable_pct >= 45 }
        set_global_variable = { name = riseandfall_story_world_pressure value = -1 }
        set_global_variable = { name = riseandfall_story_world_climate value = 3 }
    }
    else = {
        set_global_variable = { name = riseandfall_story_world_climate value = 2 }
    }
}

riseandfall_clear_story_director_vars_se = {
    # Hard reset used when Story Mode is off or when a ruler should no longer
    # carry stale director state. These variables are recalculated yearly, so it
    # is safer to clear them than to let old story roles linger after land loss.
    remove_variable = riseandfall_story_role
    remove_variable = riseandfall_story_role_years
    remove_variable = riseandfall_story_protection
    remove_variable = riseandfall_story_collapse_pressure
    remove_variable = riseandfall_story_arc_cooldown
    remove_variable = riseandfall_story_world_climate
    remove_variable = riseandfall_story_blob_pressure
    remove_variable = riseandfall_story_fatigue
    remove_variable = riseandfall_story_reign_tension
    remove_variable = riseandfall_story_stagnation_pressure
    remove_variable = riseandfall_story_ancient_regime
    remove_variable = riseandfall_story_last_succession_year
    remove_variable = riseandfall_story_no_succession_years
    remove_variable = riseandfall_story_ruler_shield
    remove_variable = riseandfall_story_complacent_vassals
    remove_variable = riseandfall_story_complacent_vassal_pressure
    remove_variable = riseandfall_story_rival_count
    remove_variable = riseandfall_story_strongest_rival_pct
    remove_variable = riseandfall_story_external_rival_relief
    remove_variable = riseandfall_story_strength_baseline
    remove_variable = riseandfall_story_collapse_gain_last
    remove_variable = riseandfall_story_no_rival_years
    remove_variable = riseandfall_story_no_rival_pressure
    remove_variable = riseandfall_rs_development_pct
}

riseandfall_apply_successor_realm_grace_se = {
    # New regimes need breathing room after inheritance, palace coups, splits,
    # and other handoffs. Grace is visible stability support plus a cooldown so
    # the director does not immediately punish the realm it just helped create.
    if = {
        limit = { is_landed = yes }
        set_variable = { name = riseandfall_story_successor_grace value = 45 }
        set_variable = { name = riseandfall_story_successor_grace_years value = 12 }
        set_variable = { name = riseandfall_story_successor_grace_decay value = 4 }
        set_variable = { name = low_stability_cooldown value = yes years = 10 }
        add_character_flag = { flag = rf_story_successor_grace days = 4380 }
        if = {
            limit = {
                has_variable = riseandfall_realm_stability_score
                NOT = { var:riseandfall_realm_stability_score >= 55 }
            }
            set_variable = { name = riseandfall_realm_stability_score value = 55 }
        }
    }
}

riseandfall_apply_palace_coup_successor_grace_se = {
    # Palace coup winners get the same scaffold as normal successors, slightly
    # stronger but faster-decaying because their legitimacy is more fragile.
    riseandfall_apply_successor_realm_grace_se = yes
    set_variable = { name = riseandfall_story_successor_grace value = 50 }
    set_variable = { name = riseandfall_story_successor_grace_decay value = 5 }
}

riseandfall_update_successor_realm_grace_se = {
    # Decay grace yearly. This creates a campaign beat: a successor has a window
    # to stabilize the realm, then the normal director logic starts judging them.
    if = {
        limit = { has_variable = riseandfall_story_successor_grace }
        if = {
            limit = { NOT = { has_variable = riseandfall_story_successor_grace_decay } }
            set_variable = { name = riseandfall_story_successor_grace_decay value = 3 }
        }
        if = {
            limit = { NOT = { has_variable = riseandfall_story_successor_grace_years } }
            set_variable = { name = riseandfall_story_successor_grace_years value = 1 }
        }
        set_variable = { name = tmp_story_successor_grace_decay value = var:riseandfall_story_successor_grace_decay }
        change_variable = { name = tmp_story_successor_grace_decay multiply = -1 }
        change_variable = { name = riseandfall_story_successor_grace add = var:tmp_story_successor_grace_decay }
        change_variable = { name = riseandfall_story_successor_grace_years add = -1 }
        if = {
            limit = {
                OR = {
                    NOT = { var:riseandfall_story_successor_grace > 0 }
                    NOT = { var:riseandfall_story_successor_grace_years > 0 }
                }
            }
            remove_variable = riseandfall_story_successor_grace
            remove_variable = riseandfall_story_successor_grace_years
            remove_variable = riseandfall_story_successor_grace_decay
            remove_variable = tmp_story_successor_grace_decay
            remove_character_flag = rf_story_successor_grace
        }
        else = {
            remove_variable = tmp_story_successor_grace_decay
        }
    }
}

riseandfall_mark_story_succession_year_se = {
    # Succession resets long-reign pressure. The director treats ruler turnover
    # as a natural release valve, not just as a generic stability shock.
    set_variable = { name = riseandfall_story_last_succession_year value = current_year }
    set_variable = { name = riseandfall_story_no_succession_years value = 0 }
    remove_variable = riseandfall_story_reign_tension
    remove_variable = riseandfall_story_stagnation_pressure
    remove_variable = riseandfall_story_ancient_regime
}

riseandfall_inherit_story_collapse_buildup_se = {
    # Collapse buildup is pressure attached to the realm's story, not only the
    # dying ruler. Preserve the predecessor's accumulated meter so inheritance
    # cannot wipe a long-running collapse arc.
    if = {
        limit = {
            exists = scope:predecessor
            OR = {
                has_game_rule = riseandfall_dynamic_story_mode_dynamic
                has_game_rule = riseandfall_dynamic_story_mode_forced
            }
        }
        if = {
            limit = {
                scope:predecessor = { has_variable = riseandfall_story_collapse_pressure }
                OR = {
                    NOT = { has_variable = riseandfall_story_collapse_pressure }
                    var:riseandfall_story_collapse_pressure < scope:predecessor.var:riseandfall_story_collapse_pressure
                }
            }
            set_variable = { name = riseandfall_story_collapse_pressure value = scope:predecessor.var:riseandfall_story_collapse_pressure }
            clamp_variable = { name = riseandfall_story_collapse_pressure min = 0 max = 100 }
        }
        if = {
            limit = {
                scope:predecessor = { has_variable = riseandfall_story_no_rival_years }
                OR = {
                    NOT = { has_variable = riseandfall_story_no_rival_years }
                    var:riseandfall_story_no_rival_years < scope:predecessor.var:riseandfall_story_no_rival_years
                }
            }
            set_variable = { name = riseandfall_story_no_rival_years value = scope:predecessor.var:riseandfall_story_no_rival_years }
            clamp_variable = { name = riseandfall_story_no_rival_years min = 0 max = 100 }
        }
        if = {
            limit = { has_variable = riseandfall_story_collapse_pressure }
            set_variable = { name = riseandfall_story_collapse_gain_last value = 0 }
        }
    }
}

riseandfall_apply_dynamic_story_pressure_se = {
    # Per-ruler campaign director. This effect converts the global climate and
    # local realm story into a transparent stability component:
    # positive values protect fragile actors, negative values pressure isolated
    # oversized realms, and successor grace remains separately visible.
    set_variable = { name = riseandfall_rs_story_pressure_component value = 0 }
    set_variable = { name = riseandfall_story_recovery_component value = 0 }
    set_variable = { name = riseandfall_story_protection value = 0 }
    set_variable = { name = riseandfall_story_blob_pressure value = 0 }
    set_variable = { name = riseandfall_story_stagnation_pressure value = 0 }
    set_variable = { name = riseandfall_story_ancient_regime value = 0 }
    set_variable = { name = riseandfall_story_ruler_shield value = 0 }
    set_variable = { name = riseandfall_story_complacent_vassals value = 0 }
    set_variable = { name = riseandfall_story_complacent_vassal_pressure value = 0 }
    set_variable = { name = riseandfall_story_rival_count value = 0 }
    set_variable = { name = riseandfall_story_strongest_rival_pct value = 0 }
    set_variable = { name = riseandfall_story_external_rival_relief value = 0 }
    set_variable = { name = riseandfall_story_collapse_gain_last value = 0 }
    set_variable = { name = riseandfall_story_no_rival_pressure value = 0 }

    if = {
        limit = { exists = global_var:riseandfall_story_world_climate }
        set_variable = { name = riseandfall_story_world_climate value = global_var:riseandfall_story_world_climate }
    }
    else = {
        set_variable = { name = riseandfall_story_world_climate value = 2 }
    }

    if = {
        limit = {
            OR = {
                has_game_rule = riseandfall_dynamic_story_mode_off
                NOT = {
                    OR = {
                        has_game_rule = riseandfall_dynamic_story_mode_dynamic
                        has_game_rule = riseandfall_dynamic_story_mode_forced
                    }
                }
            }
        }
        riseandfall_clear_story_director_vars_se = yes
    }
    else_if = {
        limit = {
            is_independent_ruler = yes
            primary_title.tier >= tier_kingdom
        }
        if = { limit = { NOT = { has_variable = riseandfall_story_role_years } } set_variable = { name = riseandfall_story_role_years value = 0 } }
        change_variable = { name = riseandfall_story_role_years add = 1 }

        # Use share of tracked world realm size instead of raw realm_size. Raw size varies too much between total conversion mods.
        set_variable = { name = tmp_story_world_total value = global_var:riseandfall_world_total }
        if = { limit = { var:tmp_story_world_total < 1 } set_variable = { name = tmp_story_world_total value = 1 } }
        set_variable = { name = tmp_story_world_development_total value = global_var:riseandfall_world_total_development }
        if = { limit = { var:tmp_story_world_development_total < 1 } set_variable = { name = tmp_story_world_development_total value = 1 } }
        set_variable = { name = tmp_story_realm_pct value = realm_size }
        change_variable = { name = tmp_story_realm_pct multiply = 100 }
        change_variable = { name = tmp_story_realm_pct divide = var:tmp_story_world_total }
        round_variable = { name = tmp_story_realm_pct nearest = 1 }
        if = {
            limit = { NOT = { has_variable = riseandfall_rs_total_development } }
            set_variable = { name = riseandfall_rs_total_development value = 0 }
            save_scope_as = ruler
            every_realm_county = {
                scope:ruler = {
                    change_variable = { name = riseandfall_rs_total_development add = prev.development_level }
                }
            }
        }
        set_variable = { name = tmp_story_development_pct value = var:riseandfall_rs_total_development }
        change_variable = { name = tmp_story_development_pct multiply = 100 }
        change_variable = { name = tmp_story_development_pct divide = var:tmp_story_world_development_total }
        round_variable = { name = tmp_story_development_pct nearest = 1 }
        set_variable = { name = riseandfall_rs_development_pct value = var:tmp_story_development_pct }
        set_variable = { name = tmp_story_stress_score value = 0 }
        set_variable = { name = tmp_story_blob_score value = 0 }
        # Rival strength is dynamic: measure the strongest nearby non-allied
        # kingdom+ realm as a percent of this ruler's integrated strength. During
        # civil wars, use the last non-civil-war baseline so rebel vassals do not
        # make the empire look artificially balanced against outside rivals.
        set_variable = { name = tmp_story_ruler_strength value = max_military_strength }
        if = {
            limit = { is_in_civil_war = no }
            set_variable = { name = riseandfall_story_strength_baseline value = var:tmp_story_ruler_strength }
        }
        else_if = {
            limit = { NOT = { has_variable = riseandfall_story_strength_baseline } }
            set_variable = { name = riseandfall_story_strength_baseline value = var:tmp_story_ruler_strength }
        }
        else_if = {
            limit = { var:riseandfall_story_strength_baseline > var:tmp_story_ruler_strength }
            set_variable = { name = tmp_story_ruler_strength value = var:riseandfall_story_strength_baseline }
        }
        if = { limit = { var:tmp_story_ruler_strength < 1 } set_variable = { name = tmp_story_ruler_strength value = 1 } }
        save_scope_as = ruler

        # Local danger thresholds mirror the low-stability event curve. Story Mode
        # should not treat a hegemony at 48 stability as "fine" just because a
        # kingdom would not yet be below 40.
        set_variable = { name = tmp_story_low_threshold value = 40 }
        if = { limit = { primary_title.tier >= tier_empire } set_variable = { name = tmp_story_low_threshold value = 45 } }
        if = { limit = { primary_title.tier >= tier_hegemony } set_variable = { name = tmp_story_low_threshold value = 50 } }
        set_variable = { name = tmp_story_stable_threshold value = var:tmp_story_low_threshold }
        change_variable = { name = tmp_story_stable_threshold add = 25 }
        set_variable = { name = tmp_story_decline_threshold value = var:tmp_story_low_threshold }
        change_variable = { name = tmp_story_decline_threshold add = 20 }

        # Find the strongest nearby rival before assigning roles. Distant powers
        # and allies do not count; only a border or close across-water rival gives
        # vassals a reason to tolerate the imperial center.
        every_neighboring_and_across_water_top_liege_realm_owner = {
            limit = {
                is_landed = yes
                primary_title.tier >= tier_kingdom
                NOT = { this = scope:ruler }
                NOT = { is_allied_to = scope:ruler }
            }
            save_scope_as = story_rival_candidate
            scope:ruler = {
                change_variable = { name = riseandfall_story_rival_count add = 1 }
                set_variable = { name = tmp_story_candidate_rival_strength value = scope:story_rival_candidate.max_military_strength }
                set_variable = { name = tmp_story_candidate_rival_pct value = var:tmp_story_candidate_rival_strength }
                change_variable = { name = tmp_story_candidate_rival_pct multiply = 100 }
                change_variable = { name = tmp_story_candidate_rival_pct divide = var:tmp_story_ruler_strength }
                if = {
                    limit = { var:tmp_story_candidate_rival_pct > var:riseandfall_story_strongest_rival_pct }
                    set_variable = { name = riseandfall_story_strongest_rival_pct value = var:tmp_story_candidate_rival_pct }
                }
            }
        }
        clamp_variable = { name = riseandfall_story_strongest_rival_pct min = 0 max = 150 }
        round_variable = { name = riseandfall_story_strongest_rival_pct nearest = 1 }

        # Convert rival strength into yearly collapse-gain relief. This is a
        # pressure reducer, not immunity: 35% strength => about 1.8 relief/year,
        # 60% => 3, 100% => 5, and superior rivals cap at 7.5.
        set_variable = { name = riseandfall_story_external_rival_relief value = var:riseandfall_story_strongest_rival_pct }
        change_variable = { name = riseandfall_story_external_rival_relief divide = 20 }
        clamp_variable = { name = riseandfall_story_external_rival_relief min = 0 max = 7.5 }

        # Story roles are selected as candidates each pass, then held for a short
        # arc so rulers do not bounce between Rival, Wildcard, and Background
        # every year. Successor grace and active collapse buildup can still force
        # immediate role changes because they represent explicit story state.
        # 1 anchor, 2 rival, 3 declining power, 4 wildcard, 5 background.
        # Declining power is based on realm scale and stability. External rivals
        # no longer hard-block it; they dynamically reduce collapse buildup later.
        set_variable = { name = tmp_story_candidate_role value = 5 }
        if = {
            limit = { has_variable = riseandfall_story_role }
            set_variable = { name = tmp_story_previous_role value = var:riseandfall_story_role }
        }
        else = {
            set_variable = { name = tmp_story_previous_role value = 0 }
        }
        if = {
            limit = { has_variable = riseandfall_story_successor_grace }
            set_variable = { name = tmp_story_candidate_role value = 5 }
        }
        else_if = {

            else_if = {
                limit = { has_game_rule = riseandfall_dynamic_story_mode_forced }
                clamp_variable = { name = tmp_story_collapse_gain min = 0 max = 14 }
            }
            # Store the net yearly meter change for the GUI. The director-note
            # rows show raw warning scores, but this value is what actually gets
            # added to collapse buildup after scaling, rival relief, and caps.
            set_variable = { name = riseandfall_story_collapse_gain_last value = var:tmp_story_collapse_gain }
            change_variable = { name = riseandfall_story_collapse_pressure add = var:tmp_story_collapse_gain }
            clamp_variable = { name = riseandfall_story_collapse_pressure min = 0 max = 100 }
        }
        else_if = {
            limit = {
                OR = {
                    var:riseandfall_story_stagnation_pressure > 0
                    var:riseandfall_story_complacent_vassal_pressure > 0
                    var:riseandfall_story_no_rival_pressure > 0
                }
                OR = {
                    var:riseandfall_story_strongest_rival_pct < 60
                    var:riseandfall_story_no_rival_pressure > 0
                    AND = {
                        var:riseandfall_story_complacent_vassal_pressure > 4
                        var:riseandfall_story_stagnation_pressure > 4
                    }
                }
                NOT = { has_variable = riseandfall_story_successor_grace }
                var:tmp_story_blob_score >= 2
            }
            if = { limit = { NOT = { has_variable = riseandfall_story_collapse_pressure } } set_variable = { name = riseandfall_story_collapse_pressure value = 0 } }
            set_variable = { name = tmp_story_collapse_gain value = 0 }
            if = {
                limit = { var:riseandfall_story_stagnation_pressure > 0 }
                set_variable = { name = tmp_story_scaled_stagnation value = var:riseandfall_story_stagnation_pressure }
                change_variable = { name = tmp_story_scaled_stagnation divide = 5 }
                change_variable = { name = tmp_story_collapse_gain add = var:tmp_story_scaled_stagnation }
            }
            if = {
                limit = { var:riseandfall_story_complacent_vassal_pressure > 0 }
                set_variable = { name = tmp_story_scaled_vassal_pressure value = var:riseandfall_story_complacent_vassal_pressure }
                change_variable = { name = tmp_story_scaled_vassal_pressure divide = 4 }
                change_variable = { name = tmp_story_collapse_gain add = var:tmp_story_scaled_vassal_pressure }
            }
            if = {
                limit = { var:riseandfall_story_no_rival_pressure > 0 }
                set_variable = { name = tmp_story_scaled_rival_pressure value = var:riseandfall_story_no_rival_pressure }
                change_variable = { name = tmp_story_scaled_rival_pressure divide = 4 }
                change_variable = { name = tmp_story_collapse_gain add = var:tmp_story_scaled_rival_pressure }
            }
            if = { limit = { has_game_rule = riseandfall_dynamic_story_mode_forced } change_variable = { name = tmp_story_collapse_gain add = 1 } }
            if = {
                limit = { var:riseandfall_story_external_rival_relief > 0 }
                set_variable = { name = tmp_story_external_rival_relief_negative value = var:riseandfall_story_external_rival_relief }
                change_variable = { name = tmp_story_external_rival_relief_negative multiply = -1 }
                change_variable = { name = tmp_story_collapse_gain add = var:tmp_story_external_rival_relief_negative }
            }
            if = {
                limit = {
                    var:riseandfall_story_strongest_rival_pct < 60
                    var:tmp_story_collapse_gain < 1
                }
                set_variable = { name = tmp_story_collapse_gain value = 1 }
            }
            # Same split as above, but for softer indirect pressure from old
            # reigns, bored vassals, and lack of rivals. Forced mode lets these
            # background tensions become collapse pressure more quickly.
            if = {
                limit = { has_game_rule = riseandfall_dynamic_story_mode_dynamic }
                clamp_variable = { name = tmp_story_collapse_gain min = 0 max = 5 }
            }
            else_if = {
                limit = { has_game_rule = riseandfall_dynamic_story_mode_forced }
                clamp_variable = { name = tmp_story_collapse_gain min = 0 max = 8 }
            }
            # Store the softer indirect-pressure gain separately so the GUI can
            # explain why raw warnings like vassal ambition do not add 1:1.
            set_variable = { name = riseandfall_story_collapse_gain_last value = var:tmp_story_collapse_gain }
            change_variable = { name = riseandfall_story_collapse_pressure add = var:tmp_story_collapse_gain }
            clamp_variable = { name = riseandfall_story_collapse_pressure min = 0 max = 100 }
        }
        else_if = {
            limit = { has_variable = riseandfall_story_collapse_pressure }
            set_variable = { name = riseandfall_story_collapse_gain_last value = -3 }
            change_variable = { name = riseandfall_story_collapse_pressure add = -3 }
            if = { limit = { NOT = { var:riseandfall_story_collapse_pressure > 0 } } remove_variable = riseandfall_story_collapse_pressure }
        }

        # Convert internal collapse pressure into the visible stability penalty:
        # Dynamic: pressure 100 => -20. This is a narrative nudge: enough to make
        # a complacent blob vulnerable, but not usually enough to delete it alone.
        # Forced: pressure 100 => -100. This is an overt collapse lever for games
        # where the director should reliably break stagnant superstates.
        if = {
            limit = { has_variable = riseandfall_story_collapse_pressure }
            set_variable = { name = riseandfall_story_blob_pressure value = var:riseandfall_story_collapse_pressure }
            change_variable = { name = riseandfall_story_blob_pressure divide = 5 }
            change_variable = { name = riseandfall_story_blob_pressure multiply = -1 }
            if = { limit = { has_game_rule = riseandfall_dynamic_story_mode_forced } change_variable = { name = riseandfall_story_blob_pressure multiply = 5 } }
            clamp_variable = { name = riseandfall_story_blob_pressure min = -100 max = 0 }
        }

        if = {
            limit = {
                global_var:riseandfall_story_world_pressure < 0
                has_variable = riseandfall_realm_stability_score
                NOT = { var:riseandfall_realm_stability_score >= 50 }
            }
            if = { limit = { NOT = { var:riseandfall_realm_stability_score >= 25 } } set_variable = { name = riseandfall_story_recovery_component value = 25 } }
            else_if = { limit = { NOT = { var:riseandfall_realm_stability_score >= 40 } } set_variable = { name = riseandfall_story_recovery_component value = 18 } }
            else = { set_variable = { name = riseandfall_story_recovery_component value = 8 } }
        }

        # Collapse targets should not receive Story Mode help from another
        # director lane. Once buildup exists, or while the realm is explicitly a
        # Declining Power, only the negative collapse pressure should apply.
        if = {
            limit = {
                OR = {
                    has_variable = riseandfall_story_collapse_pressure
                    var:riseandfall_story_role = 3
                }
            }
            set_variable = { name = riseandfall_story_protection value = 0 }
            set_variable = { name = riseandfall_story_recovery_component value = 0 }
        }

        set_variable = { name = riseandfall_rs_story_pressure_component value = 0 }
        change_variable = { name = riseandfall_rs_story_pressure_component add = var:riseandfall_story_protection }
        change_variable = { name = riseandfall_rs_story_pressure_component add = var:riseandfall_story_recovery_component }
        change_variable = { name = riseandfall_rs_story_pressure_component add = var:riseandfall_story_blob_pressure }
        if = {
            limit = {
                has_variable = riseandfall_realm_stability_score
                var:riseandfall_realm_stability_score >= 70
                var:riseandfall_rs_story_pressure_component > 0
            }
            set_variable = { name = riseandfall_rs_story_pressure_component value = 0 }
        }
        clamp_variable = { name = riseandfall_rs_story_pressure_component min = -100 max = 35 }

        remove_variable = tmp_story_world_total
        remove_variable = tmp_story_world_development_total
        remove_variable = tmp_story_realm_pct
        remove_variable = tmp_story_development_pct
        remove_variable = tmp_story_stress_score
        remove_variable = tmp_story_blob_score
        remove_variable = tmp_story_ruler_strength
        if = { limit = { has_variable = tmp_story_candidate_rival_strength } remove_variable = tmp_story_candidate_rival_strength }
        if = { limit = { has_variable = tmp_story_candidate_rival_pct } remove_variable = tmp_story_candidate_rival_pct }
        remove_variable = tmp_story_low_threshold
        remove_variable = tmp_story_stable_threshold
        remove_variable = tmp_story_decline_threshold
        remove_variable = tmp_story_candidate_role
        remove_variable = tmp_story_previous_role
        if = { limit = { has_variable = tmp_story_isolation_year_gain } remove_variable = tmp_story_isolation_year_gain }
        if = { limit = { has_variable = tmp_story_external_rival_relief_negative } remove_variable = tmp_story_external_rival_relief_negative }
        if = { limit = { has_variable = tmp_story_collapse_gain } remove_variable = tmp_story_collapse_gain }
        if = { limit = { has_variable = tmp_story_scaled_stagnation } remove_variable = tmp_story_scaled_stagnation }
        if = { limit = { has_variable = tmp_story_scaled_vassal_pressure } remove_variable = tmp_story_scaled_vassal_pressure }
        if = { limit = { has_variable = tmp_story_scaled_rival_pressure } remove_variable = tmp_story_scaled_rival_pressure }
        remove_variable = tmp_story_last_succession_year
    }
    else = {
        riseandfall_clear_story_director_vars_se = yes
    }
}
