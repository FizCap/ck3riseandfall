##########################
# Rise and Fall - Realm Split System
# Dynamic realm split: seeds breakaways by kingdom first, then allows extra duchy-level fractures in large realms.
# Breakaways become independent tributaries while the original ruler keeps their capital core and the unassigned remainder.
# Supports governments without powerful vassals (Celestial, Japanese, Administrative, Meritocratic, etc.)
# All governments use county-tier or above

riseandfall_mark_primary_breakaway_seed_se = {
    add_character_flag = { flag = rf_breakaway_ruler days = 30 }
    set_variable = { name = rf_seeded_kingdom value = scope:rf_seed_kingdom }
    set_variable = { name = rf_seeded_duchy value = capital_county.duchy }
    set_variable = { name = rf_primary_kingdom_seed value = 1 }
    scope:original_ruler = { change_variable = { name = rf_breakaway_seed_count add = 1 } }
}

riseandfall_mark_secondary_breakaway_seed_se = {
    add_character_flag = { flag = rf_breakaway_ruler days = 30 }
    set_variable = { name = rf_seeded_kingdom value = scope:rf_seed_kingdom }
    set_variable = { name = rf_seeded_duchy value = capital_county.duchy }
    scope:original_ruler = { change_variable = { name = rf_breakaway_seed_count add = 1 } }
}

##########################
# Main Realm Split Effect
# Called on the ruler who is being split

riseandfall_realm_split_se = {
    save_scope_as = original_ruler
    primary_title = { save_scope_as = original_primary }

    scope:original_ruler = {
        set_variable = { name = rf_breakaway_seed_count value = 0 }
        capital_county.kingdom = { save_scope_as = rf_capital_kingdom }
        remove_character_flag = rf_split_small
        remove_character_flag = rf_split_medium
        remove_character_flag = rf_split_large
        remove_character_flag = rf_split_extreme
        remove_character_flag = rf_split_fracture
        remove_character_flag = rf_split_major_fracture
        remove_character_flag = rf_split_total_collapse
        remove_character_flag = rf_split_lower_tier

        every_vassal = {
            limit = { is_landed = yes }
            remove_character_flag = rf_breakaway_ruler
            remove_variable = rf_my_breakaway
            remove_variable = rf_seeded_kingdom
            remove_variable = rf_seeded_duchy
            remove_variable = rf_primary_kingdom_seed
            remove_variable = rf_execute_breakaway
        }

        if = {
            limit = { realm_size >= 20 }
            add_character_flag = rf_split_small
        }
        if = {
            limit = { realm_size >= 45 }
            add_character_flag = rf_split_medium
        }
        if = {
            limit = { realm_size >= 90 }
            add_character_flag = rf_split_large
        }
        if = {
            limit = { realm_size >= 180 }
            add_character_flag = rf_split_extreme
        }
        if = {
            limit = { realm_size >= 70 }
            add_character_flag = rf_split_fracture
        }
        if = {
            limit = { realm_size >= 120 }
            add_character_flag = rf_split_major_fracture
        }
        if = {
            limit = { realm_size >= 220 }
            add_character_flag = rf_split_total_collapse
        }
        if = {
            limit = { primary_title.tier <= tier_kingdom }
            add_character_flag = rf_split_lower_tier
        }
    }

    # Hegemony special case: make all empire-tier vassals break away automatically.
    if = {
        limit = { primary_title.tier = tier_hegemony }
        scope:original_ruler = {
            every_vassal = {
                limit = {
                    is_landed = yes
                    primary_title.tier >= tier_empire
                }
                add_character_flag = { flag = rf_breakaway_ruler days = 30 }
                this = { riseandfall_execute_realm_split_se = yes }
            }
        }
    }

    # Phase 1: one primary seed per non-capital kingdom, with the odds scaling by realm size.
    scope:original_ruler = {
        every_realm_de_jure_kingdom = {
            limit = { NOT = { this = scope:rf_capital_kingdom } }
            save_scope_as = rf_seed_kingdom

            if = {
                limit = { scope:original_ruler = { has_character_flag = rf_split_extreme } }
                scope:original_ruler = {
                    ordered_vassal = {
                        limit = {
                            is_landed = yes
                            primary_title.tier >= tier_county
                            NOT = { primary_title.tier >= tier_empire }
                            NOT = { has_character_flag = rf_breakaway_ruler }
                            capital_county.kingdom = scope:rf_seed_kingdom
                        }
                        order_by = current_military_strength
                        max = 1
                        riseandfall_mark_primary_breakaway_seed_se = yes
                    }
                }
            }
            else_if = {
                limit = { scope:original_ruler = { has_character_flag = rf_split_large } }
                random = {
                    chance = 90
                    scope:original_ruler = {
                        ordered_vassal = {
                            limit = {
                                is_landed = yes
                                primary_title.tier >= tier_county
                                NOT = { primary_title.tier >= tier_empire }
                                NOT = { has_character_flag = rf_breakaway_ruler }
                                capital_county.kingdom = scope:rf_seed_kingdom
                            }
                            order_by = current_military_strength
                            max = 1
                            riseandfall_mark_primary_breakaway_seed_se = yes
                        }
                    }
                }
            }
            else_if = {
                limit = { scope:original_ruler = { has_character_flag = rf_split_medium } }
                random = {
                    chance = 70
                    scope:original_ruler = {
                        ordered_vassal = {
                            limit = {
                                is_landed = yes
                                primary_title.tier >= tier_county
                                NOT = { primary_title.tier >= tier_empire }
                                NOT = { has_character_flag = rf_breakaway_ruler }
                                capital_county.kingdom = scope:rf_seed_kingdom
                            }
                            order_by = current_military_strength
                            max = 1
                            riseandfall_mark_primary_breakaway_seed_se = yes
                        }
                    }
                }
            }
            else_if = {
                limit = { scope:original_ruler = { has_character_flag = rf_split_small } }
                random = {
                    chance = 45
                    scope:original_ruler = {
                        ordered_vassal = {
                            limit = {
                                is_landed = yes
                                primary_title.tier >= tier_county
                                NOT = { primary_title.tier >= tier_empire }
                                NOT = { has_character_flag = rf_breakaway_ruler }
                                capital_county.kingdom = scope:rf_seed_kingdom
                            }
                            order_by = current_military_strength
                            max = 1
                            riseandfall_mark_primary_breakaway_seed_se = yes
                        }
                    }
                }
            }
            else = {
                random = {
                    chance = 20
                    scope:original_ruler = {
                        ordered_vassal = {
                            limit = {
                                is_landed = yes
                                primary_title.tier >= tier_county
                                NOT = { primary_title.tier >= tier_empire }
                                NOT = { has_character_flag = rf_breakaway_ruler }
                                capital_county.kingdom = scope:rf_seed_kingdom
                            }
                            order_by = current_military_strength
                            max = 1
                            riseandfall_mark_primary_breakaway_seed_se = yes
                        }
                    }
                }
            }
        }
    }

    # Fallback so smaller realms can still produce a clean single split.
    if = {
        limit = {
            has_variable = rf_breakaway_seed_count
            var:rf_breakaway_seed_count = 0
        }
        ordered_vassal = {
            limit = {
                is_landed = yes
                primary_title.tier >= tier_county
                NOT = { primary_title.tier >= tier_empire }
                NOT = { capital_county.duchy = scope:original_ruler.capital_county.duchy }
                NOT = { capital_county.kingdom = scope:rf_capital_kingdom }
            }
            order_by = current_military_strength
            max = 1
            capital_county.kingdom = { save_scope_as = rf_seed_kingdom }
            riseandfall_mark_primary_breakaway_seed_se = yes
        }
    }
    if = {
        limit = {
            has_variable = rf_breakaway_seed_count
            var:rf_breakaway_seed_count = 0
        }
        ordered_vassal = {
            limit = {
                is_landed = yes
                primary_title.tier >= tier_county
                NOT = { primary_title.tier >= tier_empire }
                NOT = { capital_county.duchy = scope:original_ruler.capital_county.duchy }
            }
            order_by = current_military_strength
            max = 1
            capital_county.kingdom = { save_scope_as = rf_seed_kingdom }
            riseandfall_mark_primary_breakaway_seed_se = yes
        }
    }

    # Phase 2: large realms can splinter further at duchy scale, including the
    # capital kingdom as long as the ruler keeps the capital duchy.
    scope:original_ruler = {
        every_vassal = {
            limit = {
                is_landed = yes
                OR = {
                    primary_title.tier >= tier_duchy
                    AND = {
                        scope:original_ruler = { has_character_flag = rf_split_lower_tier }
                        primary_title.tier >= tier_county
                    }
                }
                NOT = { primary_title.tier >= tier_empire }
                NOT = { has_character_flag = rf_breakaway_ruler }
                NOT = { capital_county.duchy = scope:original_ruler.capital_county.duchy }
            }
            capital_county.kingdom = { save_scope_as = rf_seed_kingdom }

            if = {
                limit = {
                    scope:original_ruler = { has_character_flag = rf_split_lower_tier }
                    scope:original_ruler = { has_character_flag = rf_split_major_fracture }
                }
                random = {
                    chance = 85
                    riseandfall_mark_secondary_breakaway_seed_se = yes
                }
            }
            else_if = {
                limit = {
                    scope:original_ruler = { has_character_flag = rf_split_lower_tier }
                    scope:original_ruler = { has_character_flag = rf_split_fracture }
                }
                random = {
                    chance = 60
                    riseandfall_mark_secondary_breakaway_seed_se = yes
                }
            }
            else_if = {
                limit = { scope:original_ruler = { has_character_flag = rf_split_lower_tier } }
                random = {
                    chance = 35
                    riseandfall_mark_secondary_breakaway_seed_se = yes
                }
            }
            else_if = {
                limit = { scope:original_ruler = { has_character_flag = rf_split_total_collapse } }
                random = {
                    chance = 40
                    riseandfall_mark_secondary_breakaway_seed_se = yes
                }
            }
            else_if = {
                limit = { scope:original_ruler = { has_character_flag = rf_split_major_fracture } }
                random = {
                    chance = 24
                    riseandfall_mark_secondary_breakaway_seed_se = yes
                }
            }
            else_if = {
                limit = { scope:original_ruler = { has_character_flag = rf_split_fracture } }
                random = {
                    chance = 14
                    riseandfall_mark_secondary_breakaway_seed_se = yes
                }
            }
        }
    }

    # Phase 3: seed cores always pull their own duchy first.
    scope:original_ruler = {
        every_vassal = {
            limit = {
                has_character_flag = rf_breakaway_ruler
                has_variable = rf_seeded_duchy
            }
            save_scope_as = rf_active_seed

            scope:original_ruler = {
                every_vassal = {
                    limit = {
                        is_landed = yes
                        NOT = { has_character_flag = rf_breakaway_ruler }
                        NOT = { has_variable = rf_my_breakaway }
                        primary_title.tier >= tier_county
                        NOT = { primary_title.tier >= tier_empire }
                        NOT = { capital_county.duchy = scope:original_ruler.capital_county.duchy }
                        capital_county.duchy = scope:rf_active_seed.var:rf_seeded_duchy
                    }
                    set_variable = { name = rf_my_breakaway value = scope:rf_active_seed }
                }
            }
        }
    }

    # Phase 4: secondary seeds can peel off adjacent pockets inside their kingdom.
    scope:original_ruler = {
        every_vassal = {
            limit = {
                has_character_flag = rf_breakaway_ruler
                has_variable = rf_seeded_kingdom
                NOT = { has_variable = rf_primary_kingdom_seed }
            }
            save_scope_as = rf_active_seed

            scope:original_ruler = {
                every_vassal = {
                    limit = {
                        is_landed = yes
                        NOT = { has_character_flag = rf_breakaway_ruler }
                        NOT = { has_variable = rf_my_breakaway }
                        primary_title.tier >= tier_county
                        NOT = { primary_title.tier >= tier_empire }
                        NOT = { capital_county.duchy = scope:original_ruler.capital_county.duchy }
                        capital_county.kingdom = scope:rf_active_seed.var:rf_seeded_kingdom
                        primary_title = { is_neighbor_to_realm = scope:rf_active_seed }
                    }

                    if = {
                        limit = {
                            scope:original_ruler = { has_character_flag = rf_split_lower_tier }
                            scope:original_ruler = { has_character_flag = rf_split_major_fracture }
                        }
                        random = {
                            chance = 75
                            set_variable = { name = rf_my_breakaway value = scope:rf_active_seed }
                        }
                    }
                    else_if = {
                        limit = {
                            scope:original_ruler = { has_character_flag = rf_split_lower_tier }
                            scope:original_ruler = { has_character_flag = rf_split_fracture }
                        }
                        random = {
                            chance = 55
                            set_variable = { name = rf_my_breakaway value = scope:rf_active_seed }
                        }
                    }
                    else_if = {
                        limit = { scope:original_ruler = { has_character_flag = rf_split_lower_tier } }
                        random = {
                            chance = 35
                            set_variable = { name = rf_my_breakaway value = scope:rf_active_seed }
                        }
                    }
                    else_if = {
                        limit = { scope:original_ruler = { has_character_flag = rf_split_total_collapse } }
                        random = {
                            chance = 55
                            set_variable = { name = rf_my_breakaway value = scope:rf_active_seed }
                        }
                    }
                    else_if = {
                        limit = { scope:original_ruler = { has_character_flag = rf_split_major_fracture } }
                        random = {
                            chance = 38
                            set_variable = { name = rf_my_breakaway value = scope:rf_active_seed }
                        }
                    }
                    else_if = {
                        limit = { scope:original_ruler = { has_character_flag = rf_split_fracture } }
                        random = {
                            chance = 24
                            set_variable = { name = rf_my_breakaway value = scope:rf_active_seed }
                        }
                    }
                }
            }
        }
    }

    # Phase 5: primary kingdom seeds absorb the rest of their kingdom, keeping borders readable.
    scope:original_ruler = {
        every_vassal = {
            limit = {
                has_character_flag = rf_breakaway_ruler
                has_variable = rf_seeded_kingdom
                has_variable = rf_primary_kingdom_seed
                scope:original_primary = {
                    OR = {
                        tier = tier_empire
                        tier = tier_hegemony
                    }
                }
            }
            save_scope_as = rf_active_seed

            scope:original_ruler = {
                every_vassal = {
                    limit = {
                        is_landed = yes
                        NOT = { has_character_flag = rf_breakaway_ruler }
                        NOT = { has_variable = rf_my_breakaway }
                        primary_title.tier >= tier_county
                        NOT = { primary_title.tier >= tier_empire }
                        NOT = { capital_county.duchy = scope:original_ruler.capital_county.duchy }
                        capital_county.kingdom = scope:rf_active_seed.var:rf_seeded_kingdom
                    }
                    set_variable = { name = rf_my_breakaway value = scope:rf_active_seed }
                }
            }
        }
    }

    # Phase 6: let adjacent seeds occasionally pull border marches for a less tidy finish,
    # including inside the capital kingdom outside the capital duchy.
    scope:original_ruler = {
        every_vassal = {
            limit = { has_character_flag = rf_breakaway_ruler }
            save_scope_as = rf_active_seed

            scope:original_ruler = {
                every_vassal = {
                    limit = {
                        is_landed = yes
                        NOT = { has_character_flag = rf_breakaway_ruler }
                        NOT = { has_variable = rf_my_breakaway }
                        primary_title.tier >= tier_county
                        NOT = { primary_title.tier >= tier_empire }
                        NOT = { capital_county.duchy = scope:original_ruler.capital_county.duchy }
                        primary_title = { is_neighbor_to_realm = scope:rf_active_seed }
                    }
                    if = {
                        limit = { scope:original_ruler = { has_character_flag = rf_split_lower_tier } }
                        random = {
                            chance = 60
                            set_variable = { name = rf_my_breakaway value = scope:rf_active_seed }
                        }
                    }
                    else = {
                        random = {
                            chance = 35
                            set_variable = { name = rf_my_breakaway value = scope:rf_active_seed }
                        }
                    }
                }
            }
        }
    }

    # Phase 7: tiny breakaways are absorbed into stronger nearby successors so
    # the result is messier and less full of one-county shards.
    scope:original_ruler = {
        every_vassal = {
            limit = {
                has_character_flag = rf_breakaway_ruler
                NOT = { primary_title.tier >= tier_kingdom }
                realm_size < 5
            }
            save_scope_as = rf_small_seed

            scope:original_ruler = {
                ordered_vassal = {
                    limit = {
                        has_character_flag = rf_breakaway_ruler
                        NOT = { this = scope:rf_small_seed }
                        capital_county.kingdom = scope:rf_small_seed.capital_county.kingdom
                    }
                    order_by = current_military_strength
                    max = 1
                    scope:rf_small_seed = {
                        set_variable = { name = rf_my_breakaway value = prev }
                        remove_character_flag = rf_breakaway_ruler
                        remove_variable = rf_seeded_kingdom
                        remove_variable = rf_seeded_duchy
                        remove_variable = rf_primary_kingdom_seed
                    }
                }
            }

            if = {
                limit = {
                    NOT = { has_character_flag = rf_breakaway_ruler }
                    has_variable = rf_my_breakaway
                }
            }
            else = {
                scope:original_ruler = {
                    ordered_vassal = {
                        limit = {
                            has_character_flag = rf_breakaway_ruler
                            NOT = { this = scope:rf_small_seed }
                            primary_title = { is_neighbor_to_realm = scope:rf_small_seed }
                        }
                        order_by = current_military_strength
                        max = 1
                        scope:rf_small_seed = {
                            set_variable = { name = rf_my_breakaway value = prev }
                            remove_character_flag = rf_breakaway_ruler
                            remove_variable = rf_seeded_kingdom
                            remove_variable = rf_seeded_duchy
                            remove_variable = rf_primary_kingdom_seed
                        }
                    }
                }
            }
        }
    }

    # Execute all marked breakaways without relying on named slots.
    scope:original_ruler = {
        every_vassal = {
            limit = { has_character_flag = rf_breakaway_ruler }
            set_variable = { name = rf_execute_breakaway value = 1 }
        }
        every_vassal = {
            limit = { has_variable = rf_execute_breakaway }
            remove_variable = rf_execute_breakaway
            this = { riseandfall_execute_realm_split_se = yes }
        }
    }

    # Cleanup temporary variables and add cooldown.
    scope:original_ruler = {
        remove_variable = rf_breakaway_seed_count
        remove_character_flag = rf_split_small
        remove_character_flag = rf_split_medium
        remove_character_flag = rf_split_large
        remove_character_flag = rf_split_extreme
        remove_character_flag = rf_split_fracture
        remove_character_flag = rf_split_major_fracture
        remove_character_flag = rf_split_total_collapse
        remove_character_flag = rf_split_lower_tier
    }
    set_variable = { name = rf_realm_split_cooldown value = 1 years = 5 }
}

##########################
# Execute Realm Split
# Called on a breakaway ruler - makes them independent with their assigned vassals

riseandfall_execute_realm_split_se = {
    save_scope_as = new_realm_ruler
    
    # Pick a random naming style: 1 = Capital name, 2 = Ruler name
    random_list = {
        50 = { set_variable = { name = rf_title_style value = 1 } } # Capital name (e.g. "Constantinople Domain")
        50 = { set_variable = { name = rf_title_style value = 2 } } # Ruler name (e.g. "Li's Realm")
    }
    
    # Create the new realm title
    # Note: Hegemony tier cannot have dynamic titles, so treat it as empire
    scope:original_ruler = {
        primary_title = {
            if = {
                limit = {
                    OR = {
                        tier = tier_empire
                        tier = tier_hegemony
                    }
                }
                scope:new_realm_ruler = {
                    save_scope_as = rf_breakaway_ruler
                    capital_county = { save_scope_as = rf_base_county }
                    capital_province = { save_scope_as = rf_base_province }

                    # If the breakaway already holds an empire-tier primary title, use it
                    if = {
                        limit = { scope:rf_breakaway_ruler.primary_title.tier >= tier_empire }
                        scope:rf_breakaway_ruler = {
                            primary_title = {
                                save_scope_as = rf_new_title
                                set_color_from_title = scope:rf_base_county
                                # Set coat of arms from dynasty if ruler has one
                                if = {
                                    limit = { scope:rf_breakaway_ruler = { has_dynasty = yes } }
                                    set_coa = scope:rf_breakaway_ruler.dynasty
                                }
                            }
                        }
                    }
                    else = {
                        if = {
                            limit = {
                                has_variable = rf_title_style
                                var:rf_title_style = 1
                            }
                            create_dynamic_title = {
                                tier = empire
                                name = RISEANDFALL_BREAKAWAY_NAME_CAPITAL
                            }
                        }
                        else = {
                            create_dynamic_title = {
                                tier = empire
                                name = RISEANDFALL_BREAKAWAY_NAME_RULER
                            }
                        }
                        if = {
                            limit = { exists = scope:new_title }
                            scope:new_title = {
                                save_scope_as = rf_new_title
                                set_color_from_title = scope:rf_base_county
                                # Set coat of arms from dynasty if ruler has one
                                if = {
                                    limit = { scope:rf_breakaway_ruler = { has_dynasty = yes } }
                                    set_coa = scope:rf_breakaway_ruler.dynasty
                                }
                            }
                        }
                    }
                }
            }
            else_if = {
                limit = { tier = tier_kingdom }
                scope:new_realm_ruler = {
                    save_scope_as = rf_breakaway_ruler
                    capital_county = { save_scope_as = rf_base_county }
                    capital_province = { save_scope_as = rf_base_province }
                    if = {
                        limit = { scope:rf_breakaway_ruler.primary_title.tier >= tier_kingdom }
                        scope:rf_breakaway_ruler = {
                            primary_title = {
                                save_scope_as = rf_new_title
                                set_color_from_title = scope:rf_base_county
                                if = {
                                    limit = { scope:rf_breakaway_ruler = { has_dynasty = yes } }
                                    set_coa = scope:rf_breakaway_ruler.dynasty
                                }
                            }
                        }
                    }
                    else = {
                        if = {
                            limit = {
                                has_variable = rf_title_style
                                var:rf_title_style = 1
                            }
                            create_dynamic_title = {
                                tier = kingdom
                                name = RISEANDFALL_BREAKAWAY_NAME_CAPITAL
                            }
                        }
                        else = {
                            create_dynamic_title = {
                                tier = kingdom
                                name = RISEANDFALL_BREAKAWAY_NAME_RULER
                            }
                        }
                        if = {
                            limit = { exists = scope:new_title }
                            scope:new_title = {
                                save_scope_as = rf_new_title
                                set_color_from_title = scope:rf_base_county
                                if = {
                                    limit = { scope:rf_breakaway_ruler = { has_dynasty = yes } }
                                    set_coa = scope:rf_breakaway_ruler.dynasty
                                }
                            }
                        }
                    }
                }
            }
            else = {
                scope:new_realm_ruler = {
                    save_scope_as = rf_breakaway_ruler
                    capital_county = { save_scope_as = rf_base_county }
                    capital_province = { save_scope_as = rf_base_province }
                    if = {
                        limit = { scope:rf_breakaway_ruler.primary_title.tier >= tier_duchy }
                        scope:rf_breakaway_ruler = {
                            primary_title = {
                                save_scope_as = rf_new_title
                                set_color_from_title = scope:rf_base_county
                                if = {
                                    limit = { scope:rf_breakaway_ruler = { has_dynasty = yes } }
                                    set_coa = scope:rf_breakaway_ruler.dynasty
                                }
                            }
                        }
                    }
                    else = {
                        if = {
                            limit = {
                                has_variable = rf_title_style
                                var:rf_title_style = 1
                            }
                            create_dynamic_title = {
                                tier = duchy
                                name = RISEANDFALL_BREAKAWAY_NAME_CAPITAL
                            }
                        }
                        else = {
                            create_dynamic_title = {
                                tier = duchy
                                name = RISEANDFALL_BREAKAWAY_NAME_RULER
                            }
                        }
                        if = {
                            limit = { exists = scope:new_title }
                            scope:new_title = {
                                save_scope_as = rf_new_title
                                set_color_from_title = scope:rf_base_county
                                if = {
                                    limit = { scope:rf_breakaway_ruler = { has_dynasty = yes } }
                                    set_coa = scope:rf_breakaway_ruler.dynasty
                                }
                            }
                        }
                    }
                }
            }
        }
    }
    
    # Cleanup the style variable
    remove_variable = rf_title_style
    
    # Give them the new title
    if = {
        limit = { exists = scope:rf_new_title }
        create_title_and_vassal_change = {
            type = created
            save_scope_as = rf_title_change
            add_claim_on_loss = yes
        }
        scope:rf_new_title = {
            change_title_holder = {
                holder = scope:new_realm_ruler
                change = scope:rf_title_change
            }
        }
        resolve_title_and_vassal_change = scope:rf_title_change

        # original ruler should get an unpressed claim on the brand new breakaway title
        scope:original_ruler = {
            add_unpressed_claim = scope:rf_new_title
        }
    }
    
    # Transfer assigned vassals
    create_title_and_vassal_change = {
        type = conquest
        save_scope_as = rf_vassal_change
        add_claim_on_loss = no
    }
    
    scope:original_ruler = {
        every_vassal = {
            limit = {
                has_variable = rf_my_breakaway
                var:rf_my_breakaway = scope:new_realm_ruler
            }
            change_liege = {
                liege = scope:new_realm_ruler
                change = scope:rf_vassal_change
            }
            remove_variable = rf_my_breakaway
        }
    }
    
    resolve_title_and_vassal_change = scope:rf_vassal_change
    
    # Make independent
    create_title_and_vassal_change = {
        type = independency
        save_scope_as = rf_indep_change
        add_claim_on_loss = yes
    }
    
    becomes_independent = {
        change = scope:rf_indep_change
    }
    
    resolve_title_and_vassal_change = scope:rf_indep_change
    
    # Make tributary of original ruler
    start_tributary = {
        contract_group = tributary_settled
        suzerain = scope:original_ruler
    }

    # Add claim on original primary title
    # If the original primary was a hegemon, ensure the original ruler
    # gets an unpressed claim on the breakaway's (possibly newly-created)
    # empire title so the hegemon can press for it later.
    if = {
        limit = { scope:original_primary.tier = tier_hegemony }
        if = { limit = { exists = scope:rf_new_title } scope:original_ruler = { add_unpressed_claim = scope:rf_new_title } }
        else = { if = { limit = { exists = scope:new_realm_ruler.primary_title } scope:original_ruler = { add_unpressed_claim = scope:new_realm_ruler.primary_title } } }
    }

    add_unpressed_claim = scope:original_primary

    if = {
        limit = { exists = scope:original_ruler }
        if = {
            limit = {
                can_set_relation_rival_trigger = { CHARACTER = scope:original_ruler }
                NOT = { has_relation_rival = scope:original_ruler }
            }
            set_relation_rival = { target = scope:original_ruler reason = riseandfall_realmsplit_became_rival }
        }
        scope:original_ruler = {
            if = {
                limit = {
                    can_set_relation_rival_trigger = { CHARACTER = scope:new_realm_ruler }
                    NOT = { has_relation_rival = scope:new_realm_ruler }
                }
                set_relation_rival = { target = scope:new_realm_ruler reason = riseandfall_realmsplit_became_rival }
            }
        }
    }

    if = {
        limit = { exists = scope:new_realm_ruler }
        scope:new_realm_ruler = {
            set_variable = { name = low_stability_cooldown value = yes years = 5 }
        }
    }

    # Mark this realm as a successor state and link it to its split origin title.
    # Stored both on the ruler and the successor primary title for future mechanics.
    set_variable = { name = rf_successor_origin_title value = scope:original_primary }
    add_character_flag = rf_successor_state_ruler
    if = {
        limit = { exists = scope:new_realm_ruler.primary_title }
        # Capture the initial realm size on the new realm ruler so we can store
        # a permanent snapshot on the title for future shrink checks.
        scope:rf_breakaway_ruler = { set_variable = { name = tmp_successor_realm_size value = realm_size } }
        scope:new_realm_ruler.primary_title = {
            set_variable = { name = rf_successor_origin_title value = scope:original_primary years = 50 }
            set_variable = { name = rf_is_successor_split_title value = 1 years = 50 }
            # Permanent marker for this title indicating it's a split realm successor
            set_variable = { name = rf_is_permanent_successor_split value = 1 }
            # Store the baseline realm size (number of counties) on the title permanently
            # Only set if the title doesn't already have a snapshot to avoid accidental overwrite
            if = {
                limit = { NOT = { has_variable = rf_successor_permanent_realm_size } }
                set_variable = { name = rf_successor_permanent_realm_size value = scope:rf_breakaway_ruler.var:tmp_successor_realm_size }
            }
        }
        # Clear the temporary holder variable from the ruler scope
        scope:rf_breakaway_ruler = { remove_variable = tmp_successor_realm_size }
    }

    # Ensure all successor rulers from this same split origin have mutual claims
    # on each other's primary titles (including the newly-created breakaway).
    every_independent_ruler = {
        limit = {
            NOT = { this = scope:new_realm_ruler }
            is_landed = yes
            exists = primary_title
            primary_title = {
                has_variable = rf_is_successor_split_title
                has_variable = rf_successor_origin_title
            }
            has_weak_claim_on = scope:original_primary
        }
        save_scope_as = rf_peer_successor
        scope:new_realm_ruler = { add_unpressed_claim = scope:rf_peer_successor.primary_title }
        scope:rf_peer_successor = { add_unpressed_claim = scope:new_realm_ruler.primary_title }
    }

    # If an empire-tier split throws off a very large kingdom successor, allow one
    # follow-up fracture inside that kingdom so it does not leave fully intact.
    if = {
        limit = {
            scope:original_primary = {
                OR = {
                    tier = tier_empire
                    tier = tier_hegemony
                }
            }
            primary_title.tier = tier_kingdom
            realm_size >= 18
            any_vassal = {
                is_landed = yes
                primary_title.tier >= tier_county
                NOT = { primary_title.tier >= tier_kingdom }
                NOT = { capital_county.duchy = scope:new_realm_ruler.capital_county.duchy }
            }
        }
        if = {
            limit = {
                realm_size >= 45
                any_vassal = {
                    is_landed = yes
                    primary_title.tier >= tier_duchy
                    NOT = { primary_title.tier >= tier_kingdom }
                    NOT = { capital_county.duchy = scope:new_realm_ruler.capital_county.duchy }
                }
            }
            random = {
                chance = 85
                riseandfall_realm_split_se = yes
            }
        }
        else_if = {
            limit = { realm_size >= 30 }
            random = {
                chance = 65
                riseandfall_realm_split_se = yes
            }
        }
        else = {
            random = {
                chance = 40
                riseandfall_realm_split_se = yes
            }
        }
    }


    # --- End new rivalry & cross-claims block ---

    remove_variable = rf_seeded_kingdom
    remove_variable = rf_seeded_duchy
    remove_variable = rf_primary_kingdom_seed
    remove_variable = rf_execute_breakaway

    # Remove flag
    remove_character_flag = rf_breakaway_ruler
}
