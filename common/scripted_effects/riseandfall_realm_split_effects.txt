# Rise and Fall - Realm Split System
# Kingdoms seed by duchy, empires by kingdom, and hegemonies by empire
# unless the preferred tier has too few usable regions.
# The original ruler keeps their capital core and any territory outside the seeded regions.
# Supports governments without powerful vassals (Celestial, Japanese, Administrative, Meritocratic, etc.)
# All governments use county-tier or above

riseandfall_mark_primary_breakaway_seed_se = {
    add_character_flag = { flag = rf_breakaway_ruler days = 30 }
    set_variable = { name = rf_seeded_kingdom value = scope:rf_seed_kingdom }
    set_variable = { name = rf_seeded_duchy value = capital_county.duchy }
    set_variable = { name = rf_primary_kingdom_seed value = 1 }
    scope:original_ruler = { change_variable = { name = rf_breakaway_seed_count add = 1 } }
}

riseandfall_mark_secondary_breakaway_seed_se = {
    add_character_flag = { flag = rf_breakaway_ruler days = 30 }
    set_variable = { name = rf_seeded_kingdom value = scope:rf_seed_kingdom }
    set_variable = { name = rf_seeded_duchy value = capital_county.duchy }
    scope:original_ruler = { change_variable = { name = rf_breakaway_seed_count add = 1 } }
}

riseandfall_mark_empire_breakaway_seed_se = {
    add_character_flag = { flag = rf_breakaway_ruler days = 30 }
    set_variable = { name = rf_seeded_empire value = scope:rf_seed_empire }
    set_variable = { name = rf_seeded_kingdom value = capital_county.kingdom }
    set_variable = { name = rf_seeded_duchy value = capital_county.duchy }
    set_variable = { name = rf_primary_kingdom_seed value = 1 }
    scope:original_ruler = { change_variable = { name = rf_breakaway_seed_count add = 1 } }
}

riseandfall_assign_saved_kingdom_to_seed_se = {
    scope:original_ruler = {
        every_vassal = {
            limit = {
                is_landed = yes
                NOT = { has_character_flag = rf_breakaway_ruler }
                NOT = { has_character_flag = rf_palace_coup_split_protected }
                NOT = { has_variable = rf_my_breakaway }
                primary_title.tier >= tier_county
                NOT = { primary_title.tier >= tier_empire }
                NOT = { capital_county.duchy = scope:original_ruler.capital_county.duchy }
                capital_county.kingdom = scope:rf_assign_kingdom
            }
            set_variable = { name = rf_my_breakaway value = scope:rf_active_seed }
        }
    }
}

riseandfall_assign_saved_duchy_to_seed_se = {
    scope:original_ruler = {

            }

}

riseandfall_score_breakaway_influence_for_vassal_se = {
    save_scope_as = rf_candidate_breakaway
    scope:rf_influence_vassal = {
        set_variable = { name = rf_sphere_score value = 0 }

        if = {
            limit = {
                scope:rf_candidate_breakaway = { has_variable = rf_seeded_empire }
                capital_county.kingdom = { de_jure_liege = scope:rf_candidate_breakaway.var:rf_seeded_empire }
            }
            change_variable = { name = rf_sphere_score add = 160 }
        }
        if = {
            limit = {
                scope:rf_candidate_breakaway = { has_variable = rf_seeded_kingdom }
                capital_county.kingdom = scope:rf_candidate_breakaway.var:rf_seeded_kingdom
            }
            change_variable = { name = rf_sphere_score add = 140 }
        }
        if = {
            limit = {
                scope:rf_candidate_breakaway = { has_variable = rf_seeded_duchy }
                capital_county.duchy = scope:rf_candidate_breakaway.var:rf_seeded_duchy
            }
            change_variable = { name = rf_sphere_score add = 120 }
        }
        if = {
            limit = { scope:rf_candidate_breakaway.primary_title = { is_neighbor_to_realm = scope:rf_influence_vassal } }
            change_variable = { name = rf_sphere_score add = 85 }
        }
        if = {
            limit = { primary_title = { is_neighbor_to_realm = scope:rf_candidate_breakaway } }
            change_variable = { name = rf_sphere_score add = 65 }
        }
        if = {
            limit = { is_allied_to = scope:rf_candidate_breakaway }
            change_variable = { name = rf_sphere_score add = 100 }
        }
        if = {
            limit = { opinion = { target = scope:rf_candidate_breakaway value >= 60 } }
            change_variable = { name = rf_sphere_score add = 70 }
        }
        else_if = {
            limit = { opinion = { target = scope:rf_candidate_breakaway value >= 20 } }
            change_variable = { name = rf_sphere_score add = 35 }
        }
        else_if = {
            limit = { opinion = { target = scope:rf_candidate_breakaway value <= -40 } }
            change_variable = { name = rf_sphere_score add = -55 }
        }
        if = {
            limit = { has_same_culture_as = scope:rf_candidate_breakaway }
            change_variable = { name = rf_sphere_score add = 35 }
        }
        if = {
            limit = { faith = scope:rf_candidate_breakaway.faith }
            change_variable = { name = rf_sphere_score add = 35 }
        }
        else_if = {
            limit = { religion = scope:rf_candidate_breakaway.religion }
            change_variable = { name = rf_sphere_score add = 15 }
        }
        if = {
            limit = {
                scope:rf_candidate_breakaway = { has_dynasty = yes }
                has_dynasty = yes
                dynasty = scope:rf_candidate_breakaway.dynasty
            }
            change_variable = { name = rf_sphere_score add = 45 }
        }
        if = {
            limit = { has_claim_on = scope:original_primary }
            change_variable = { name = rf_sphere_score add = 25 }
        }
        if = {
            limit = { current_military_strength > scope:rf_candidate_breakaway.current_military_strength }
            change_variable = { name = rf_sphere_score add = 20 }
        }
        if = {
            limit = {
                scope:rf_candidate_breakaway = { has_character_flag = rf_major_breakaway_seed }
            }
            change_variable = { name = rf_sphere_score add = 25 }
        }
        if = {
            limit = {
                scope:rf_candidate_breakaway = { has_character_flag = rf_secondary_breakaway_seed }
            }
            change_variable = { name = rf_sphere_score add = 15 }
        }
        if = {
            limit = {
                has_variable = rf_best_sphere_score
                var:rf_sphere_score > var:rf_best_sphere_score
            }
            set_variable = { name = rf_best_sphere_score value = var:rf_sphere_score }
            set_variable = { name = rf_my_breakaway value = scope:rf_candidate_breakaway }
        }
    }
}

riseandfall_assign_vassal_to_best_breakaway_sphere_se = {
    save_scope_as = rf_influence_vassal
    set_variable = { name = rf_best_sphere_score value = -9999 }

    scope:original_ruler = {
        every_vassal = {
            limit = {
                is_landed = yes
                has_character_flag = rf_breakaway_ruler
                NOT = { this = scope:rf_influence_vassal }
            }
            riseandfall_score_breakaway_influence_for_vassal_se = yes
        }
    }

    remove_variable = rf_sphere_score
    remove_variable = rf_best_sphere_score
}

# Main Realm Split Effect
# Called on the ruler who is being split

riseandfall_realm_split_se = {
    save_scope_as = original_ruler
    primary_title = { save_scope_as = original_primary }

    scope:original_ruler = {
        set_variable = { name = rf_breakaway_seed_count value = 0 }
        set_variable = { name = rf_usable_empire_anchor_count value = 0 }
        set_variable = { name = rf_usable_kingdom_anchor_count value = 0 }
        set_variable = { name = rf_usable_duchy_anchor_count value = 0 }
        capital_county.kingdom = { save_scope_as = rf_capital_kingdom }
        capital_county.duchy = { save_scope_as = rf_capital_duchy }
        capital_county.kingdom = {
            if = {
                limit = { exists = de_jure_liege }
                de_jure_liege = { save_scope_as = rf_capital_empire }
            }
        }
        remove_character_flag = rf_split_small
        remove_character_flag = rf_split_medium
        remove_character_flag = rf_split_large
        remove_character_flag = rf_split_extreme
        remove_character_flag = rf_split_fracture
        remove_character_flag = rf_split_major_fracture
        remove_character_flag = rf_split_total_collapse
        remove_character_flag = rf_split_lower_tier
        remove_character_flag = rf_split_seed_by_empire
        remove_character_flag = rf_split_seed_by_kingdom
        remove_character_flag = rf_split_seed_by_duchy

        every_vassal = {
            limit = { is_landed = yes }
            remove_character_flag = rf_breakaway_ruler
            remove_character_flag = rf_major_breakaway_seed
            remove_character_flag = rf_secondary_breakaway_seed
            remove_variable = rf_my_breakaway
            remove_variable = rf_seeded_empire
            remove_variable = rf_seeded_kingdom
            remove_variable = rf_seeded_duchy
            remove_variable = rf_primary_kingdom_seed
            remove_variable = rf_execute_breakaway
        }

        every_vassal = {
            limit = {
                is_landed = yes
                has_character_flag = rf_forced_breakaway_seed
            }
            add_character_flag = { flag = rf_breakaway_ruler days = 30 }
            capital_county.kingdom = { save_scope_as = rf_seed_kingdom }
            set_variable = { name = rf_seeded_kingdom value = scope:rf_seed_kingdom }
            set_variable = { name = rf_seeded_duchy value = capital_county.duchy }
            scope:original_ruler = { change_variable = { name = rf_breakaway_seed_count add = 1 } }
        }

        every_realm_de_jure_empire = {
            limit = {
                NOT = { this = scope:rf_capital_empire }
            }
            scope:original_ruler = { change_variable = { name = rf_usable_empire_anchor_count add = 1 } }
        }
        every_realm_de_jure_kingdom = {
            limit = {
                NOT = { this = scope:rf_capital_kingdom }
            }
            scope:original_ruler = { change_variable = { name = rf_usable_kingdom_anchor_count add = 1 } }
        }
        every_realm_de_jure_duchy = {
            limit = {
                NOT = { this = scope:rf_capital_duchy }
            }
            scope:original_ruler = { change_variable = { name = rf_usable_duchy_anchor_count add = 1 } }
        }

        if = {
            limit = {
                primary_title.tier = tier_hegemony
                has_variable = rf_usable_empire_anchor_count
                var:rf_usable_empire_anchor_count >= 2
            }
            add_character_flag = rf_split_seed_by_empire
        }
        else_if = {
            limit = {
                primary_title.tier >= tier_empire
                has_variable = rf_usable_kingdom_anchor_count
                var:rf_usable_kingdom_anchor_count >= 2
            }
            add_character_flag = rf_split_seed_by_kingdom
        }
        else = {
            add_character_flag = rf_split_seed_by_duchy
        }

        if = {
            limit = {
                any_vassal = {
                    is_landed = yes
                    has_trait = warlord
                }
            }
            every_vassal = {
                limit = { is_landed = yes }
                remove_trait = warlord
            }
        }

        if = {
            limit = { realm_size >= 20 }
            add_character_flag = rf_split_small
        }
        if = {
            limit = { realm_size >= 45 }
            add_character_flag = rf_split_medium
        }
        if = {
            limit = { realm_size >= 90 }
            add_character_flag = rf_split_large
        }
        if = {
            limit = { realm_size >= 180 }
            add_character_flag = rf_split_extreme
        }
        if = {
            limit = { realm_size >= 70 }
            add_character_flag = rf_split_fracture
        }
        if = {
            limit = { realm_size >= 120 }
            add_character_flag = rf_split_major_fracture
        }
        if = {
            limit = { realm_size >= 220 }
            add_character_flag = rf_split_total_collapse
        }
        if = {
            limit = { primary_title.tier <= tier_kingdom }
            add_character_flag = rf_split_lower_tier
        }
    }

    # Hegemony special case: make all empire-tier vassals break away automatically.
    if = {
        limit = { primary_title.tier = tier_hegemony }
        scope:original_ruler = {
            every_vassal = {

        }
    }
    set_variable = { name = rf_realm_split_cooldown value = 1 years = 5 }
}

# Execute Realm Split
# Called on a breakaway ruler - makes them independent with their assigned vassals

riseandfall_execute_realm_split_se = {
    save_scope_as = new_realm_ruler
    
    # Match the original ruler's rank for breakaway dynamic titles.
    # Hegemony splits fall back to empire-tier dynamic titles.
    scope:new_realm_ruler = {
        save_scope_as = rf_breakaway_ruler
        capital_county = { save_scope_as = rf_base_county }
        capital_barony = { save_scope_as = rf_base_capital_barony }

        if = {
            limit = {
                OR = {
                    scope:original_primary = { tier = tier_hegemony }
                    scope:original_primary = { tier = tier_empire }
                }
            }
            if = {
                limit = { highest_held_title_tier >= tier_empire }
                primary_title = { save_scope_as = rf_new_title }
            }
            else = {
                create_dynamic_title = {
                    tier = empire
                    name = RISEANDFALL_BREAKAWAY_NAME_CAPITAL
                }
                if = {
                    limit = { exists = scope:new_title }
                    add_character_flag = { flag = rf_created_breakaway_dynamic_title days = 7 }
                    scope:new_title = { save_scope_as = rf_new_title }
                }
            }
        }
        else_if = {
            limit = {
                scope:original_primary = { tier = tier_kingdom }
            }
            if = {
                limit = { highest_held_title_tier >= tier_kingdom }
                primary_title = { save_scope_as = rf_new_title }
            }
            else = {
                create_dynamic_title = {
                    tier = kingdom
                    name = RISEANDFALL_BREAKAWAY_NAME_CAPITAL
                }
                if = {
                    limit = { exists = scope:new_title }
                    add_character_flag = { flag = rf_created_breakaway_dynamic_title days = 7 }
                    scope:new_title = { save_scope_as = rf_new_title }
                }
            }
        }
        else_if = {
            limit = {
                scope:original_primary = { tier = tier_duchy }
            }
            if = {
                limit = { highest_held_title_tier >= tier_duchy }
                primary_title = { save_scope_as = rf_new_title }
            }
            else = {
                create_dynamic_title = {
                    tier = duchy
                    name = RISEANDFALL_BREAKAWAY_NAME_CAPITAL
                }
                if = {
                    limit = { exists = scope:new_title }
                    add_character_flag = { flag = rf_created_breakaway_dynamic_title days = 7 }
                    scope:new_title = { save_scope_as = rf_new_title }
                }
            }
        }
        else_if = {
            limit = { exists = primary_title }
            primary_title = { save_scope_as = rf_new_title }
        }
    }

    if = {
        limit = { exists = scope:rf_new_title }
        scope:rf_new_title = {
            set_color_from_title = scope:rf_base_county
            generate_coa = yes
        }
    }
    
    
    # Give them the new title
    if = {
        limit = {
            exists = scope:rf_new_title
            scope:new_realm_ruler = { has_character_flag = rf_created_breakaway_dynamic_title }
        }
        create_title_and_vassal_change = {
            type = created
            save_scope_as = rf_title_change
            add_claim_on_loss = yes
        }
        scope:rf_new_title = {
            change_title_holder = {
                holder = scope:new_realm_ruler
                change = scope:rf_title_change
            }
        }
        resolve_title_and_vassal_change = scope:rf_title_change
    }

    # CK3 may keep a reserve lower title as primary after title handoff; force the
    # same-rank successor title primary before successor war markers are applied.
    if = {
        limit = { exists = scope:rf_new_title }
        scope:new_realm_ruler = {
            set_primary_title_to = scope:rf_new_title
        }
    }
    
    # Transfer assigned vassals
    create_title_and_vassal_change = {
        type = conquest
        save_scope_as = rf_vassal_change
        add_claim_on_loss = no
    }
    
    scope:original_ruler = {
        every_vassal = {
            limit = {
                has_variable = rf_my_breakaway
                var:rf_my_breakaway = scope:new_realm_ruler
            }
            change_liege = {
                liege = scope:new_realm_ruler
                change = scope:rf_vassal_change
            }
            remove_variable = rf_my_breakaway
            remove_character_flag = rf_palace_coup_neutral_rejector
            remove_character_flag = rf_palace_coup_neutral_seed
        }
    }
    
    resolve_title_and_vassal_change = scope:rf_vassal_change
    
    # Make independent
    create_title_and_vassal_change = {
        type = independency
        save_scope_as = rf_indep_change
        add_claim_on_loss = yes
    }
    
    becomes_independent = {
        change = scope:rf_indep_change
    }
    
    resolve_title_and_vassal_change = scope:rf_indep_change
    
    # Make tributary of original ruler unless this split should create fully
    # independent successor states.
    if = {
        limit = {
            NOT = {
                scope:original_ruler = {
                    has_variable = rf_realm_split_no_tributaries
                }
            }
        }
        start_tributary = {
            contract_group = tributary_settled
            suzerain = scope:original_ruler
        }
    }

    # Claims on titles removed; successor war CB handles claims now.

    if = {
        limit = { exists = scope:new_realm_ruler }
        scope:new_realm_ruler = {
            set_variable = { name = low_stability_cooldown value = yes years = 5 }
        }
    }

    # Mark this realm as a successor state and link it to its split origin title.
    # Stored both on the ruler and the successor primary title for future mechanics.
    set_variable = { name = rf_successor_origin_title value = scope:original_primary }
    # Seed a propagated "root origin" pointer so descendant splits can be
    # associated with the original root of a split chain (enables multi-gen CBs).
    set_variable = { name = rf_successor_root_origin_title value = scope:original_primary }
    add_character_flag = rf_successor_state_ruler
    riseandfall_apply_successor_realm_grace_se = yes
    if = {
        limit = { exists = scope:new_realm_ruler.primary_title }
        # Capture the initial realm size on the new realm ruler so we can store
        # a permanent snapshot on the title for future shrink checks.
        scope:rf_breakaway_ruler = { set_variable = { name = tmp_successor_realm_size value = realm_size } }
        scope:new_realm_ruler.primary_title = {
            # immediate origin (the title that was split)
            set_variable = { name = rf_successor_origin_title value = scope:original_primary years = 50 }
            set_variable = { name = rf_is_successor_split_title value = 1 years = 30 }
            # Permanent marker for this title indicating it's a split realm successor
            set_variable = { name = rf_is_permanent_successor_split value = 1 }
            # Propagate and/or set the root origin: if the original_primary was
            # itself a successor, inherit its root; otherwise use original_primary.
            if = {
                limit = { scope:original_primary = { has_variable = rf_successor_root_origin_title } }
                set_variable = { name = rf_successor_root_origin_title value = scope:original_primary.var:rf_successor_root_origin_title years = 30 }
            }
            else = {
                set_variable = { name = rf_successor_root_origin_title value = scope:original_primary years = 30 }
            }
            # Store the baseline realm size (number of counties) on the title permanently
            # Only set if the title doesn't already have a snapshot to avoid accidental overwrite
            if = {
                limit = { NOT = { has_variable = rf_successor_permanent_realm_size } }
                set_variable = { name = rf_successor_permanent_realm_size value = scope:rf_breakaway_ruler.var:tmp_successor_realm_size }
            }
        }
        # Clear the temporary holder variable from the ruler scope
        scope:rf_breakaway_ruler = { remove_variable = tmp_successor_realm_size }
    }

    # Ensure all successor rulers from this same split origin have mutual claims
    # on each other's primary titles (including the newly-created breakaway).
    every_independent_ruler = {
        limit = {
            NOT = { this = scope:new_realm_ruler }
            is_landed = yes
            exists = primary_title
            primary_title = {
                has_variable = rf_is_successor_split_title
                has_variable = rf_successor_origin_title
            }
            has_weak_claim_on = scope:original_primary
        }
        save_scope_as = rf_peer_successor
    }

    # --- End new rivalry & cross-claims block ---

    remove_variable = rf_seeded_empire
    remove_variable = rf_seeded_kingdom
    remove_variable = rf_seeded_duchy
    remove_variable = rf_primary_kingdom_seed
    remove_variable = rf_execute_breakaway
    remove_character_flag = rf_major_breakaway_seed
    remove_character_flag = rf_secondary_breakaway_seed

    # Remove flag
    remove_character_flag = rf_breakaway_ruler
    remove_character_flag = rf_forced_breakaway_seed
    remove_character_flag = rf_created_breakaway_dynamic_title
    remove_character_flag = rf_palace_coup_neutral_rejector
    remove_character_flag = rf_palace_coup_neutral_seed
}

riseandfall_realm_split_lower_actor_primary_below_recipient_se = {
    clear_saved_scope = rf_absorbed_split_title
    clear_saved_scope = rf_absorbed_replacement_title

    if = {
        limit = {
            exists = scope:recipient
            exists = primary_title
            scope:recipient = { exists = primary_title }
            primary_title = {
                has_variable = rf_is_successor_split_title
                has_variable = rf_successor_root_origin_title
            }
            primary_title.tier >= scope:recipient.primary_title.tier
        }
        primary_title = { save_scope_as = rf_absorbed_split_title }

        # Demotion needs a real lower title to preserve the actor as a landed
        # vassal after their successor crown is destroyed or handed upward.
        if = {
            limit = { scope:recipient = { primary_title.tier >= tier_hegemony } }
            random_held_title = {
                limit = {
                    tier = tier_empire
                    is_titular = no
                    NOT = { this = scope:rf_absorbed_split_title }
                }
                save_scope_as = rf_absorbed_replacement_title
            }
        }
        if = {
            limit = {
                NOT = { exists = scope:rf_absorbed_replacement_title }
                scope:recipient = { primary_title.tier >= tier_empire }
            }
            random_held_title = {
                limit = {
                    tier = tier_kingdom
                    is_titular = no
                    NOT = { this = scope:rf_absorbed_split_title }
                }
                save_scope_as = rf_absorbed_replacement_title
            }
        }
        if = {
            limit = {
                NOT = { exists = scope:rf_absorbed_replacement_title }
                scope:recipient = { primary_title.tier >= tier_kingdom }
            }
            random_held_title = {
                limit = {
                    tier = tier_duchy
                    is_titular = no
                    NOT = { this = scope:rf_absorbed_split_title }
                }
                save_scope_as = rf_absorbed_replacement_title
            }
        }
        if = {
            limit = {
                NOT = { exists = scope:rf_absorbed_replacement_title }
                scope:recipient = { primary_title.tier >= tier_duchy }
            }
            random_held_title = {
                limit = {
                    tier = tier_county
                    is_titular = no
                    NOT = { this = scope:rf_absorbed_split_title }
                }
                save_scope_as = rf_absorbed_replacement_title
            }
        }

        if = {
            limit = { exists = scope:rf_absorbed_replacement_title }
            scope:rf_absorbed_replacement_title = {
                if = {
                    limit = { scope:rf_absorbed_split_title = { has_variable = rf_successor_origin_title } }
                    set_variable = { name = rf_successor_origin_title value = scope:rf_absorbed_split_title.var:rf_successor_origin_title years = 50 }
                }
                if = {
                    limit = { scope:rf_absorbed_split_title = { has_variable = rf_successor_root_origin_title } }
                    set_variable = { name = rf_successor_root_origin_title value = scope:rf_absorbed_split_title.var:rf_successor_root_origin_title years = 30 }
                }
                if = {
                    limit = { scope:rf_absorbed_split_title = { has_variable = rf_successor_permanent_realm_size } }
                    set_variable = { name = rf_successor_permanent_realm_size value = scope:rf_absorbed_split_title.var:rf_successor_permanent_realm_size }
                }
                else_if = {
                    limit = { NOT = { has_variable = rf_successor_permanent_realm_size } }
                    set_variable = { name = rf_successor_permanent_realm_size value = root.realm_size }
                }
                set_variable = { name = rf_is_successor_split_title value = 1 years = 30 }
                set_variable = { name = rf_is_permanent_successor_split value = 1 }
            }
            set_primary_title_to = scope:rf_absorbed_replacement_title
            scope:rf_absorbed_split_title = {
                remove_variable = rf_is_successor_split_title
                remove_variable = rf_successor_origin_title
                remove_variable = rf_successor_root_origin_title
                remove_variable = rf_is_permanent_successor_split
                remove_variable = rf_successor_permanent_realm_size
            }
            if = {
                limit = { scope:rf_absorbed_split_title = { is_titular = no } }
                # Real equal-rank crowns are submitted to the new liege. This
                # lets an empire-tier successor join another empire without
                # creating a stronger dynamic title for the recipient.
                create_title_and_vassal_change = {
                    type = granted
                    save_scope_as = rf_absorbed_title_change
                    add_claim_on_loss = no
                }
                scope:rf_absorbed_split_title = {
                    change_title_holder = {
                        holder = scope:recipient
                        change = scope:rf_absorbed_title_change
                        take_baronies = no
                    }
                }
                resolve_title_and_vassal_change = scope:rf_absorbed_title_change
                scope:recipient = {
                    if = {
                        limit = { primary_title.tier <= scope:rf_absorbed_split_title.tier }
                        set_primary_title_to = scope:rf_absorbed_split_title
                    }
                }
            }
            else = {
                # Titular successor shells have no de jure land to preserve, so
                # they are simply removed once the actor has a real title below.
                destroy_title = scope:rf_absorbed_split_title
            }
        }
    }
}

riseandfall_split_title_monitor_cleanup_se = {
    save_scope_as = rf_ruler
    capital_barony = { save_scope_as = rf_base_capital_barony }

    # Only operate on primary titles that were flagged as permanent split successors.
    if = {
        limit = {
            primary_title = {
                has_variable = rf_is_permanent_successor_split
                has_variable = rf_successor_permanent_realm_size
            }
        }
        primary_title = { save_scope_as = rf_split_title }
        clear_saved_scope = rf_replacement_split_title

        # Legacy safety repair: older palace-coup/split paths could leave the
        # successor marker on a lower-rank primary. Promote those states back to
        # the stored root origin rank, matching normal realm split behavior.
        if = {
            limit = {
                scope:rf_split_title = {
                    has_variable = rf_successor_root_origin_title
                    var:rf_successor_root_origin_title = {
                        OR = {
                            tier = tier_hegemony
                            tier = tier_empire
                        }
                    }
                    tier < tier_empire
                }
            }
            random_held_title = {
                limit = {
                    tier = tier_empire
                    NOT = { this = scope:rf_split_title }
                }
                save_scope_as = rf_replacement_split_title
            }
            if = {
                limit = { NOT = { exists = scope:rf_replacement_split_title } }
                create_dynamic_title = {
                    tier = empire
                    name = RISEANDFALL_BREAKAWAY_NAME_CAPITAL
                }
                if = {
                    limit = { exists = scope:new_title }
                    scope:new_title = { save_scope_as = rf_replacement_split_title }
                    create_title_and_vassal_change = {
                        type = created
                        save_scope_as = rf_replacement_title_change
                        add_claim_on_loss = no
                    }
                    scope:rf_replacement_split_title = {
                        change_title_holder = {
                            holder = scope:rf_ruler
                            change = scope:rf_replacement_title_change
                        }
                        generate_coa = yes
                    }
                    resolve_title_and_vassal_change = scope:rf_replacement_title_change
                }
            }
        }
        else_if = {
            limit = {
                OR = {
                    scope:rf_split_title = {
                        has_variable = rf_successor_root_origin_title
                        var:rf_successor_root_origin_title = { tier = tier_kingdom }
                        tier < tier_kingdom
                    }
                    scope:rf_split_title = {
                        NOT = { has_variable = rf_successor_root_origin_title }
                        tier < tier_kingdom
                    }
                }
            }
            random_held_title = {
                limit = {
                    tier = tier_kingdom
                    NOT = { this = scope:rf_split_title }
                }
                save_scope_as = rf_replacement_split_title
            }
            if = {
                limit = { NOT = { exists = scope:rf_replacement_split_title } }
                create_dynamic_title = {
                    tier = kingdom
                    name = RISEANDFALL_BREAKAWAY_NAME_CAPITAL
                }
                if = {
                    limit = { exists = scope:new_title }
                    scope:new_title = { save_scope_as = rf_replacement_split_title }
                    create_title_and_vassal_change = {
                        type = created
                        save_scope_as = rf_replacement_title_change
                        add_claim_on_loss = no
                    }
                    scope:rf_replacement_split_title = {
                        change_title_holder = {
                            holder = scope:rf_ruler
                            change = scope:rf_replacement_title_change
                        }
                        generate_coa = yes
                    }
                    resolve_title_and_vassal_change = scope:rf_replacement_title_change
                }
            }
        }

        if = {
            limit = {
                exists = scope:rf_replacement_split_title
                NOT = { scope:rf_replacement_split_title = scope:rf_split_title }
            }
            scope:rf_replacement_split_title = {
                if = {
                    limit = { scope:rf_split_title = { has_variable = rf_successor_origin_title } }
                    set_variable = { name = rf_successor_origin_title value = scope:rf_split_title.var:rf_successor_origin_title years = 50 }
                }
                if = {
                    limit = { scope:rf_split_title = { has_variable = rf_successor_root_origin_title } }
                    set_variable = { name = rf_successor_root_origin_title value = scope:rf_split_title.var:rf_successor_root_origin_title years = 30 }
                }
                set_variable = { name = rf_is_successor_split_title value = 1 years = 30 }
                set_variable = { name = rf_is_permanent_successor_split value = 1 }
                set_variable = { name = rf_successor_permanent_realm_size value = scope:rf_split_title.var:rf_successor_permanent_realm_size }
            }
            scope:rf_split_title = {
                remove_variable = rf_is_successor_split_title
                remove_variable = rf_successor_origin_title
                remove_variable = rf_successor_root_origin_title
                remove_variable = rf_is_permanent_successor_split
                remove_variable = rf_successor_permanent_realm_size
            }
            if = {
                limit = { scope:rf_split_title = { is_titular = yes } }
                destroy_title = scope:rf_split_title
            }
            if = {
                limit = {
                    exists = scope:rf_replacement_split_title
                    primary_title.tier <= scope:rf_replacement_split_title.tier
                }
                set_primary_title_to = scope:rf_replacement_split_title
            }
        }

        if = {
            limit = {
                is_independent_ruler = no
                scope:rf_split_title = { is_titular = yes }
            }
            destroy_title = scope:rf_split_title
        }
        else = {
            # Compute current realm size and compare to the stored snapshot on the title.
            set_variable = { name = rf_current_realm_size value = realm_size }
            if = {
                limit = {
                    primary_title = {
                        has_variable = rf_successor_permanent_realm_size
                    }
                }
                set_variable = { name = rf_snapshot value = primary_title.var:rf_successor_permanent_realm_size }
            }
            else = {
                set_variable = { name = rf_snapshot value = var:rf_current_realm_size }
            }
            change_variable = { name = rf_snapshot multiply = 0.25 }
            if = {
                limit = {
                    var:rf_current_realm_size <= var:rf_snapshot
                    # The under-25% collapse check only destroys old successor titular titles
                    # after their initial 50-year origin grace marker has expired.
                    primary_title = { NOT = { has_variable = rf_successor_origin_title } }
                }
                primary_title = { save_scope_as = rf_split_title }
                if = {
                    limit = { scope:rf_split_title = { is_titular = yes } }
                    destroy_title = scope:rf_split_title
                }
            }
            remove_variable = rf_current_realm_size
            remove_variable = rf_snapshot
        }
        clear_saved_scope = rf_replacement_split_title
    }
}
