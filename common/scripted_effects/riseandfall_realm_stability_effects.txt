##########################
# Rise and Fall - Realm Stability
# Computes a 0..100 realm stability score and stores it on the ruler as
# riseandfall_realm_stability_score
# Weighting (user spec):
#  - Ruler stats: each stat contributes (stat - 10), clamped to -6..+6.
#    The five stat contributions are then summed and clamped to -30..+30.
#  - Legitimacy: additive component (-20..+15); no legitimacy = -20, full legitimacy = +15
#  - Vassal relations: mapped to -30..+30 (average direct vassal opinion -100..100)
#    Powerful vassals count double. Noble relations dominated medieval stability.
#  - Partition succession: -10 (destabilising — historical Carolingian/West Frankish pattern)
#  - Smoothing: asymmetric — decline faster (ceil(d/3)/yr), recovery slower (ceil(d/7)/yr)

riseandfall_calculate_realm_stability_se = {
    # Root scope: called from an every_ruler on_action, so root is the ruler
    # Clear any previous working vars
    remove_variable = riseandfall_rs_stats_component
    remove_variable = riseandfall_rs_vassals_component
    remove_variable = riseandfall_rs_realm_size_component
    remove_variable = riseandfall_rs_culture_faith_component
    remove_variable = riseandfall_rs_epidemic_component
    remove_variable = riseandfall_rs_partition_component
    remove_variable = riseandfall_rs_succession_component
    remove_variable = riseandfall_rs_story_pressure_component
    remove_variable = riseandfall_story_recovery_component
    remove_variable = riseandfall_rs_strong_vassal_count
    remove_variable = riseandfall_rs_foreign_culture_county_pct
    remove_variable = riseandfall_rs_foreign_faith_county_pct
    remove_variable = riseandfall_rs_epidemic_province_pct

    # Legacy story-mode variables were removed from active use.
    # Avoid removing unset legacy vars here because CK3 logs that as a live use.

    ########################################
    # GLOBAL STAT AVERAGE CACHE (recomputed every 10 calendar years)
    # We keep track of the last year the calculation ran plus a decade counter.
    # The routine below only executes once per ingame year (using current_year)
    # and when ten years have passed we walk every_ruler and recompute averages.
    # Initialize tracking globals if missing.
    if = { limit = { NOT = { exists = global_var:riseandfall_stat_avg_last_year } } set_global_variable = { name = riseandfall_stat_avg_last_year value = 0 } }
    if = { limit = { NOT = { exists = global_var:riseandfall_stat_avg_years } } set_global_variable = { name = riseandfall_stat_avg_years value = 0 } }
    if = { limit = { NOT = { exists = global_var:riseandfall_avg_diplomacy } } set_global_variable = { name = riseandfall_avg_diplomacy value = 0 } }
    if = { limit = { NOT = { exists = global_var:riseandfall_avg_martial } } set_global_variable = { name = riseandfall_avg_martial value = 0 } }
    if = { limit = { NOT = { exists = global_var:riseandfall_avg_stewardship } } set_global_variable = { name = riseandfall_avg_stewardship value = 0 } }
    if = { limit = { NOT = { exists = global_var:riseandfall_avg_intrigue } } set_global_variable = { name = riseandfall_avg_intrigue value = 0 } }
    if = { limit = { NOT = { exists = global_var:riseandfall_avg_learning } } set_global_variable = { name = riseandfall_avg_learning value = 0 } }
    if = { limit = { NOT = { exists = global_var:riseandfall_ruler_count } } set_global_variable = { name = riseandfall_ruler_count value = 0 } }
    if = { limit = { NOT = { exists = global_var:riseandfall_avg_total } } set_global_variable = { name = riseandfall_avg_total value = 0 } }
    if = { limit = { NOT = { exists = global_var:riseandfall_world_total } } set_global_variable = { name = riseandfall_world_total value = 1 } }

    if = {
        limit = {
            OR = {
                global_var:riseandfall_avg_diplomacy <= 0
                global_var:riseandfall_avg_martial <= 0
                global_var:riseandfall_avg_stewardship <= 0
                global_var:riseandfall_avg_intrigue <= 0
                global_var:riseandfall_avg_learning <= 0
            }
        }
        set_global_variable = { name = riseandfall_stat_avg_last_year value = 0 }
    }

    # run once per year (and run an initial calculation on first call)
    if = {
        limit = { NOT = { current_year = global_var:riseandfall_stat_avg_last_year } }
        set_global_variable = { name = riseandfall_world_total value = 0 }
        every_independent_ruler = {
            change_global_variable = { name = riseandfall_world_total add = realm_size }
        }
        if = { limit = { global_var:riseandfall_world_total < 1 } set_global_variable = { name = riseandfall_world_total value = 1 } }
        # if this is the very first time we've ever executed, recompute now
        if = { limit = { global_var:riseandfall_stat_avg_last_year = 0 }
            set_global_variable = { name = riseandfall_avg_diplomacy value = 0 }
            set_global_variable = { name = riseandfall_avg_martial value = 0 }
            set_global_variable = { name = riseandfall_avg_stewardship value = 0 }
            set_global_variable = { name = riseandfall_avg_intrigue value = 0 }
            set_global_variable = { name = riseandfall_avg_learning value = 0 }
            set_global_variable = { name = riseandfall_avg_total value = 0 }
            set_global_variable = { name = riseandfall_ruler_count value = 0 }
            every_ruler = {
                change_global_variable = { name = riseandfall_ruler_count add = 1 }
                change_global_variable = { name = riseandfall_avg_diplomacy add = diplomacy }
                change_global_variable = { name = riseandfall_avg_martial add = martial }
                change_global_variable = { name = riseandfall_avg_stewardship add = stewardship }
                change_global_variable = { name = riseandfall_avg_intrigue add = intrigue }
                change_global_variable = { name = riseandfall_avg_learning add = learning }
                change_global_variable = { name = riseandfall_avg_total add = diplomacy }
                change_global_variable = { name = riseandfall_avg_total add = martial }
                change_global_variable = { name = riseandfall_avg_total add = stewardship }
                change_global_variable = { name = riseandfall_avg_total add = intrigue }
                change_global_variable = { name = riseandfall_avg_total add = learning }
            }
            if = { limit = { global_var:riseandfall_ruler_count > 0 }
                change_global_variable = { name = riseandfall_avg_diplomacy divide = global_var:riseandfall_ruler_count }
                change_global_variable = { name = riseandfall_avg_martial divide = global_var:riseandfall_ruler_count }
                change_global_variable = { name = riseandfall_avg_stewardship divide = global_var:riseandfall_ruler_count }
                change_global_variable = { name = riseandfall_avg_intrigue divide = global_var:riseandfall_ruler_count }
                change_global_variable = { name = riseandfall_avg_learning divide = global_var:riseandfall_ruler_count }
                change_global_variable = { name = riseandfall_avg_total divide = global_var:riseandfall_ruler_count }
                change_global_variable = { name = riseandfall_avg_total divide = 5 }

            }
        }
        change_global_variable = { name = riseandfall_stat_avg_years add = 1 }
        set_global_variable = { name = riseandfall_stat_avg_last_year value = current_year }
        # perform decadal recompute
        if = { limit = { global_var:riseandfall_stat_avg_years >= 10 }
            set_global_variable = { name = riseandfall_stat_avg_years value = 0 }
            set_global_variable = { name = riseandfall_avg_diplomacy value = 0 }
            set_global_variable = { name = riseandfall_avg_martial value = 0 }
            set_global_variable = { name = riseandfall_avg_stewardship value = 0 }
            set_global_variable = { name = riseandfall_avg_intrigue value = 0 }
            set_global_variable = { name = riseandfall_avg_learning value = 0 }
            set_global_variable = { name = riseandfall_avg_total value = 0 }
            set_global_variable = { name = riseandfall_ruler_count value = 0 }
            every_ruler = {
                change_global_variable = { name = riseandfall_ruler_count add = 1 }
                change_global_variable = { name = riseandfall_avg_diplomacy add = diplomacy }
                change_global_variable = { name = riseandfall_avg_martial add = martial }
                change_global_variable = { name = riseandfall_avg_stewardship add = stewardship }
                change_global_variable = { name = riseandfall_avg_intrigue add = intrigue }
                change_global_variable = { name = riseandfall_avg_learning add = learning }
                change_global_variable = { name = riseandfall_avg_total add = diplomacy }
                change_global_variable = { name = riseandfall_avg_total add = martial }
                change_global_variable = { name = riseandfall_avg_total add = stewardship }
                change_global_variable = { name = riseandfall_avg_total add = intrigue }
                change_global_variable = { name = riseandfall_avg_total add = learning }
            }
            if = { limit = { global_var:riseandfall_ruler_count > 0 }
                change_global_variable = { name = riseandfall_avg_diplomacy divide = global_var:riseandfall_ruler_count }
                change_global_variable = { name = riseandfall_avg_martial divide = global_var:riseandfall_ruler_count }
                change_global_variable = { name = riseandfall_avg_stewardship divide = global_var:riseandfall_ruler_count }
                change_global_variable = { name = riseandfall_avg_intrigue divide = global_var:riseandfall_ruler_count }
                change_global_variable = { name = riseandfall_avg_learning divide = global_var:riseandfall_ruler_count }
                # average of five stats = total / (count*5)
                change_global_variable = { name = riseandfall_avg_total divide = global_var:riseandfall_ruler_count }
                change_global_variable = { name = riseandfall_avg_total divide = 5 }

            }
        }
    }

    ########################################
    # 1) RULER STATS
    # Each stat contributes relative to a fixed baseline of 10.
    # Per-stat contribution = clamp(stat - 10, -6, +6)
    # Total stats component is the sum of Diplomacy/Martial/Stewardship/Intrigue/Learning,
    # then clamped to -30..+30.
    set_variable = { name = riseandfall_rs_stats_component value = 0 }

    # If ruler is under a regular regency, save the active diarch scope so we can read the regent's stats.
    # Puppet regencies use the ruler's own stats for realm stability calculations.
    if = {
        limit = { has_diarchy_type = regency }
        diarch = { save_scope_as = diarch }
        set_variable = { name = riseandfall_rs_use_diarch_stats value = 1 }
    }
    else = { set_variable = { name = riseandfall_rs_use_diarch_stats value = 0 } }

    # Helper macro-like repeated block for each stat: compute tmp_add = source_stat - 10,
    # clamp to -6..+6, then add to the total.
    # Diplomacy
    if = { limit = { var:riseandfall_rs_use_diarch_stats = 1 }
        set_variable = { name = tmp_add value = scope:diarch.diplomacy }
    }
    else = { set_variable = { name = tmp_add value = diplomacy } }
    change_variable = { name = tmp_add add = -10 }
    clamp_variable = { name = tmp_add min = -6 max = 6 }
    change_variable = { name = riseandfall_rs_stats_component add = var:tmp_add }

    # Martial
    if = { limit = { var:riseandfall_rs_use_diarch_stats = 1 }
        set_variable = { name = tmp_add value = scope:diarch.martial }
    }
    else = { set_variable = { name = tmp_add value = martial } }
    change_variable = { name = tmp_add add = -10 }
    clamp_variable = { name = tmp_add min = -6 max = 6 }
    change_variable = { name = riseandfall_rs_stats_component add = var:tmp_add }

    # Stewardship
    if = { limit = { var:riseandfall_rs_use_diarch_stats = 1 }
        set_variable = { name = tmp_add value = scope:diarch.stewardship }
    }
    else = { set_variable = { name = tmp_add value = stewardship } }
    change_variable = { name = tmp_add add = -10 }
    clamp_variable = { name = tmp_add min = -6 max = 6 }
    change_variable = { name = riseandfall_rs_stats_component add = var:tmp_add }

    # Intrigue
    if = { limit = { var:riseandfall_rs_use_diarch_stats = 1 }
        set_variable = { name = tmp_add value = scope:diarch.intrigue }
    }
    else = { set_variable = { name = tmp_add value = intrigue } }
    change_variable = { name = tmp_add add = -10 }
    clamp_variable = { name = tmp_add min = -6 max = 6 }
    change_variable = { name = riseandfall_rs_stats_component add = var:tmp_add }

    # Learning
    if = { limit = { var:riseandfall_rs_use_diarch_stats = 1 }
        set_variable = { name = tmp_add value = scope:diarch.learning }
    }
    else = { set_variable = { name = tmp_add value = learning } }
    change_variable = { name = tmp_add add = -10 }
    clamp_variable = { name = tmp_add min = -6 max = 6 }
    change_variable = { name = riseandfall_rs_stats_component add = var:tmp_add }

    # Clamp stats component to -30..+30 (sum of five per-stat ranges centered at 0)
    clamp_variable = { name = riseandfall_rs_stats_component min = -30 max = 30 }





    ########################################
    # 2) LEGITIMACY (additive component -20..+15)
    # Legitimacy (piety/influence/celestial) adds directly to the score.
    # No legitimacy = -20 penalty; full legitimacy = +15 bonus.
    # riseandfall_rs_legitimacy_multiplier kept at 100 (neutral) for backward compat.
    set_variable = { name = riseandfall_rs_legitimacy_component value = 0 }
    set_variable = { name = riseandfall_rs_legitimacy_multiplier value = 100 }

    # Theocracy uses piety_level (0..5). Other governments with legitimacy use
    # raw legitimacy progress against the vanilla max for their legitimacy system.
    # That keeps the mod's output in the same -20..+15 band, but scales smoothly
    # across different tiers/eras instead of snapping at legitimacy-level breakpoints.
    if = {
        limit = { has_government = theocracy_government }
        set_variable = { name = tmp_piety value = piety_level }
        if = { limit = { var:tmp_piety = 0 } set_variable = { name = tmp_piety_mapped value = -20 } }
        else_if = { limit = { var:tmp_piety = 1 } set_variable = { name = tmp_piety_mapped value = -12 } }
        else_if = { limit = { var:tmp_piety = 2 } set_variable = { name = tmp_piety_mapped value = -4 } }
        else_if = { limit = { var:tmp_piety = 3 } set_variable = { name = tmp_piety_mapped value = 4 } }
        else_if = { limit = { var:tmp_piety = 4 } set_variable = { name = tmp_piety_mapped value = 10 } }
        else = { set_variable = { name = tmp_piety_mapped value = 15 } }
        set_variable = { name = riseandfall_rs_legitimacy_component value = var:tmp_piety_mapped }
    }
    else = {
        if = {
            limit = { has_legitimacy = yes }
            set_variable = { name = tmp_leg_current value = legitimacy }
            if = {
                limit = { government_has_flag = government_is_mandala }
                set_variable = { name = tmp_leg_max value = mandala_legitimacy_max }
            }
            else_if = {
                limit = { has_government = celestial_government }
                set_variable = { name = tmp_leg_max value = mandate_legitimacy_max }
            }
            else = {
                set_variable = { name = tmp_leg_max value = legitimacy_max }
            }

            if = {
                limit = { var:tmp_leg_max > 0 }
                set_variable = { name = tmp_leg_pct value = var:tmp_leg_current }
                change_variable = { name = tmp_leg_pct divide = var:tmp_leg_max }
                change_variable = { name = tmp_leg_pct multiply = 100 }
            }
            else = {
                set_variable = { name = tmp_leg_pct value = 0 }
            }

            clamp_variable = { name = tmp_leg_pct min = 0 max = 100 }
            set_variable = { name = riseandfall_rs_legitimacy_component value = var:tmp_leg_pct }
            change_variable = { name = riseandfall_rs_legitimacy_component multiply = 0.35 }
            change_variable = { name = riseandfall_rs_legitimacy_component add = -20 }
            round_variable = { name = riseandfall_rs_legitimacy_component nearest = 1 }
        }
        else = {
            # No legitimacy mechanic available; fall back to influence_level (0..6)
            if = {
                limit = { influence_level >= 0 }
                set_variable = { name = tmp_inf value = influence_level }
                if = { limit = { var:tmp_inf = 0 } set_variable = { name = tmp_inf_mapped value = -20 } }
                else_if = { limit = { var:tmp_inf = 1 } set_variable = { name = tmp_inf_mapped value = -14 } }
                else_if = { limit = { var:tmp_inf = 2 } set_variable = { name = tmp_inf_mapped value = -7 } }
                else_if = { limit = { var:tmp_inf = 3 } set_variable = { name = tmp_inf_mapped value = 0 } }
                else_if = { limit = { var:tmp_inf = 4 } set_variable = { name = tmp_inf_mapped value = 7 } }
                else_if = { limit = { var:tmp_inf = 5 } set_variable = { name = tmp_inf_mapped value = 13 } }
                else_if = { limit = { var:tmp_inf = 6 } set_variable = { name = tmp_inf_mapped value = 15 } }
                else = { set_variable = { name = tmp_inf_mapped value = 0 } }
                set_variable = { name = riseandfall_rs_legitimacy_component value = var:tmp_inf_mapped }
            }
        }
    }

    ########################################
    # 3) VASSAL RELATIONS (-30..+30)
    # Now we compute the average vassal opinion including negatives, map -100..+100 -> -30..+30
    set_variable = { name = riseandfall_rs_vassals_component value = 0 }
    # Prepare accumulators - count vassals we actually process and sum their opinions (including negatives)
    set_variable = { name = tmp_vassal_count value = 0 }
    set_variable = { name = tmp_opinion_sum value = 0 }

    # Save ruler scope first so vassals can modify the ruler's variables
    save_scope_as = ruler

    every_vassal = {
        # For each vassal: save their opinion of the liege and add to the running sum.
        # Powerful vassals count double (both in numerator and denominator) so their
        # opinion has greater weight.
        if = {
            limit = { save_temporary_opinion_value_as = { name = tmp_vassal_opinion target = liege } }
            # If this vassal is powerful, add opinion twice and increment count by 2
            if = {
                limit = { is_powerful_vassal = yes }
                # prepare doubled opinion in a tmp variable in vassal scope
                # powerful vassal: count twice and add their opinion twice to the sum
                scope:ruler = { change_variable = { name = tmp_vassal_count add = 2 } }
                scope:ruler = { change_variable = { name = tmp_opinion_sum add = scope:tmp_vassal_opinion } }
                scope:ruler = { change_variable = { name = tmp_opinion_sum add = scope:tmp_vassal_opinion } }
            }
            else = {
                # Normal vassal: add opinion once and increment count by 1
                scope:ruler = { change_variable = { name = tmp_vassal_count add = 1 } }
                scope:ruler = { change_variable = { name = tmp_opinion_sum add = scope:tmp_vassal_opinion } }
            }
        }
    }

    # Compute average (if there are vassals) and map to -30..+30
    set_variable = { name = tmp_avg_opinion value = 0 }
    if = {
        limit = { var:tmp_vassal_count > 0 }
        set_variable = { name = tmp_avg_opinion value = var:tmp_opinion_sum }
        change_variable = { name = tmp_avg_opinion divide = var:tmp_vassal_count }
    }
    else = {
        # No vassals -> neutral opinion
        set_variable = { name = tmp_avg_opinion value = 0 }
    }

    # Map -100..+100 to -30..+30 by scaling: value * 30 / 100 (implemented as *30 / 100)
    # Noble relations were the primary driver of medieval stability; range raised from +-20 to +-30.
    # This makes an average opinion of -68 -> approx -20, +80 -> approx +24
    # Use integer math: multiply first then divide to preserve precision.
    change_variable = { name = tmp_avg_opinion multiply = 30 }
    change_variable = { name = tmp_avg_opinion divide = 100 }
    # Round the scaled value to keep the component integer
    round_variable = { name = tmp_avg_opinion nearest = 1 }
    set_variable = { name = riseandfall_rs_vassals_component value = var:tmp_avg_opinion }

    clamp_variable = { name = riseandfall_rs_vassals_component min = -30 max = 30 }

    ########################################
    # 3.6) PUPPET REGENCY TRACKING & STABILITY BONUS
    # Tracks how long a puppet_regency has been active and manages the
    # puppet stability bonus (riseandfall_rs_puppet_stability_bonus, -100..100).
    # The pool starts from the puppet master's opening stat edge vs the world
    # average, then regenerates or decays yearly like a stability mana reserve.
    if = { limit = { NOT = { has_variable = riseandfall_puppet_regency_years } } set_variable = { name = riseandfall_puppet_regency_years value = 0 } }
    if = { limit = { has_diarchy_type = puppet_regency } change_variable = { name = riseandfall_puppet_regency_years add = 1 } }
    else = { if = { limit = { var:riseandfall_puppet_regency_years > 0 } change_variable = { name = riseandfall_puppet_regency_years add = -1 } } }

    # Initialize puppet stability bonus from the starting puppet master's average stats.
    # Only set once per regency lifecycle; later years only regen or decay it.
    if = {
        limit = {
            has_diarchy_type = puppet_regency
            has_active_diarchy = yes
            exists = scope:diarch
            NOT = { has_variable = riseandfall_rs_puppet_stability_bonus }
        }
        # Compute the puppet master's average across all five stats
        set_variable = { name = tmp_pm_start_sum value = scope:diarch.diplomacy }
        change_variable = { name = tmp_pm_start_sum add = scope:diarch.martial }
        change_variable = { name = tmp_pm_start_sum add = scope:diarch.stewardship }
        change_variable = { name = tmp_pm_start_sum add = scope:diarch.intrigue }
        change_variable = { name = tmp_pm_start_sum add = scope:diarch.learning }
        change_variable = { name = tmp_pm_start_sum divide = 5 }

        # Compute delta vs precomputed world five-stat average (riseandfall_avg_total)
        set_variable = { name = tmp_pm_start_baseline value = global_var:riseandfall_avg_total }
        change_variable = { name = tmp_pm_start_baseline multiply = -1 }
        change_variable = { name = tmp_pm_start_sum add = var:tmp_pm_start_baseline }
        set_variable = { name = tmp_pm_start_delta value = var:tmp_pm_start_sum }

        # Convert the opening stat edge into starting stability mana.
        set_variable = { name = tmp_pm_start_bonus value = var:tmp_pm_start_delta }
        change_variable = { name = tmp_pm_start_bonus multiply = 5 }
        clamp_variable = { name = tmp_pm_start_bonus min = -100 max = 100 }
        round_variable = { name = tmp_pm_start_bonus nearest = 1 }

        set_variable = { name = riseandfall_rs_puppet_stability_bonus value = var:tmp_pm_start_bonus }
        clamp_variable = { name = riseandfall_rs_puppet_stability_bonus min = -100 max = 100 }
        round_variable = { name = riseandfall_rs_puppet_stability_bonus nearest = 1 }
        set_variable = { name = tmp_pm_skip_yearly_drift value = 1 }

        remove_variable = tmp_pm_start_sum
        remove_variable = tmp_pm_start_baseline
        remove_variable = tmp_pm_start_delta
        remove_variable = tmp_pm_start_bonus
    }

    # Yearly mana flow: better puppet masters rebuild stability, weak ones bleed it away.
    if = {
        limit = {
            has_diarchy_type = puppet_regency
            has_active_diarchy = yes
            exists = scope:diarch
            has_variable = riseandfall_rs_puppet_stability_bonus
            NOT = { has_variable = tmp_pm_skip_yearly_drift }
        }
        # Sum all five stats for the puppet master and compute the average
        set_variable = { name = tmp_pm_stat_sum value = scope:diarch.diplomacy }
        change_variable = { name = tmp_pm_stat_sum add = scope:diarch.martial }
        change_variable = { name = tmp_pm_stat_sum add = scope:diarch.stewardship }
        change_variable = { name = tmp_pm_stat_sum add = scope:diarch.intrigue }
        change_variable = { name = tmp_pm_stat_sum add = scope:diarch.learning }
        # Convert sum -> average (integer division)
        change_variable = { name = tmp_pm_stat_sum divide = 5 }

        # Subtract precomputed world-average-of-five-stats (riseandfall_avg_total)
        set_variable = { name = tmp_pm_avg_baseline value = global_var:riseandfall_avg_total }
        change_variable = { name = tmp_pm_avg_baseline multiply = -1 }
        change_variable = { name = tmp_pm_stat_sum add = var:tmp_pm_avg_baseline }
        # tmp_pm_stat_sum now holds (puppet_master_avg - world_avg)
        set_variable = { name = tmp_pm_stat_delta value = var:tmp_pm_stat_sum }

        # Convert the average stat edge directly into yearly mana gain/loss.
        set_variable = { name = tmp_pm_drift value = var:tmp_pm_stat_delta }
        clamp_variable = { name = tmp_pm_drift min = -10 max = 10 }
        round_variable = { name = tmp_pm_drift nearest = 1 }

        change_variable = { name = riseandfall_rs_puppet_stability_bonus add = var:tmp_pm_drift }
        clamp_variable = { name = riseandfall_rs_puppet_stability_bonus min = -100 max = 100 }
        round_variable = { name = riseandfall_rs_puppet_stability_bonus nearest = 1 }

    else_if = {

    set_variable = { name = riseandfall_rs_foreign_faith_county_pct value = 0 }

    set_variable = { name = tmp_story_strong_vassal_component_target value = 0 }

    if = {
        limit = {
            is_independent_ruler = yes

                is_landed = yes

            }
        }

        remove_variable = tmp_story_effective_liege_strength
        remove_variable = tmp_story_strong_vassal_threshold

        set_variable = { name = riseandfall_rs_strong_vassal_count value = var:tmp_story_strong_vassals }
        if = {
            limit = { var:tmp_story_strong_vassals = 0 }
            set_variable = { name = tmp_story_strong_vassal_component_target value = 10 }

        else_if = {
            limit = { var:tmp_story_strong_vassals = 1 }
            set_variable = { name = tmp_story_strong_vassal_component_target value = 0 }

        else_if = {
            limit = { var:tmp_story_strong_vassals = 2 }
            set_variable = { name = tmp_story_strong_vassal_component_target value = -10 }

            set_variable = { name = tmp_story_strong_vassal_component_target value = -20 }

                    culture = {

                            target = scope:ruler.culture

                }

            else_if = {
                limit = {
                    exists = holder
                    holder = { culture = scope:story_mode_county.culture }
                }
            }
            else = {
                scope:ruler = { change_variable = { name = tmp_story_foreign_culture_counties add = 1 } }

            if = {
                limit = { faith = scope:ruler.faith }

            else_if = {
                limit = {
                    faith = {
                        faith_hostility_level = {
                            target = scope:ruler.faith
                            value <= faith_astray_level

                    }
                }

            else_if = {
                limit = {
                    exists = holder
                    holder = { faith = scope:story_mode_county.faith }
                }
            }
            else = {
                scope:ruler = { change_variable = { name = tmp_story_foreign_faith_counties add = 1 } }

            every_county_province = {
                scope:ruler = { change_variable = { name = tmp_story_realm_provinces add = 1 } }
                if = {
                    limit = { any_province_epidemic = {} }
                    scope:ruler = { change_variable = { name = tmp_story_epidemic_provinces add = 1 } }

        if = {
            limit = { var:tmp_story_realm_counties > 0 }
            set_variable = { name = tmp_story_foreign_culture_county_pct value = var:tmp_story_foreign_culture_counties }
            change_variable = { name = tmp_story_foreign_culture_county_pct multiply = 100 }
            change_variable = { name = tmp_story_foreign_culture_county_pct divide = var:tmp_story_realm_counties }
            round_variable = { name = tmp_story_foreign_culture_county_pct nearest = 1 }

            change_variable = { name = tmp_story_foreign_faith_county_pct multiply = 100 }
            change_variable = { name = tmp_story_foreign_faith_county_pct divide = var:tmp_story_realm_counties }
            round_variable = { name = tmp_story_foreign_faith_county_pct nearest = 1 }

            set_variable = { name = riseandfall_rs_foreign_culture_county_pct value = var:tmp_story_foreign_culture_county_pct }
            set_variable = { name = riseandfall_rs_foreign_faith_county_pct value = var:tmp_story_foreign_faith_county_pct }

            round_variable = { name = riseandfall_rs_culture_faith_component nearest = 1 }
            clamp_variable = { name = riseandfall_rs_culture_faith_component min = -30 max = 10 }

        remove_variable = tmp_story_realm_counties
        remove_variable = tmp_story_foreign_culture_counties
        remove_variable = tmp_story_foreign_faith_counties
        remove_variable = tmp_story_realm_provinces
        remove_variable = tmp_story_epidemic_provinces
        remove_variable = tmp_story_foreign_culture_county_pct
        remove_variable = tmp_story_foreign_faith_county_pct
        remove_variable = tmp_story_epidemic_province_pct
        remove_variable = tmp_story_nonapproved_pct

    # Strong-vassal balance now drifts toward the actual realm state over five yearly pulses

    if = {

    }
    else = {
        set_variable = { name = tmp_rs_strong_vassal_delta value = var:tmp_story_strong_vassal_component_target }

        change_variable = { name = tmp_rs_strong_vassal_current multiply = -1 }
        change_variable = { name = tmp_rs_strong_vassal_delta add = var:tmp_rs_strong_vassal_current }

            set_variable = { name = tmp_rs_strong_vassal_delta_abs value = var:tmp_rs_strong_vassal_delta }

            set_variable = { name = tmp_rs_strong_vassal_delta_abs value = var:tmp_rs_strong_vassal_delta }
            change_variable = { name = tmp_rs_strong_vassal_delta_abs multiply = -1 }

            set_variable = { name = tmp_rs_strong_vassal_step value = 0 }

        }

        if = {
            limit = { var:tmp_rs_strong_vassal_delta > 0 }
            change_variable = { name = riseandfall_rs_strong_vassal_component add = var:tmp_rs_strong_vassal_step }
        }
        else_if = {
            limit = { var:tmp_rs_strong_vassal_delta < 0 }
            set_variable = { name = tmp_rs_strong_vassal_step_neg value = var:tmp_rs_strong_vassal_step }
            change_variable = { name = tmp_rs_strong_vassal_step_neg multiply = -1 }
            change_variable = { name = riseandfall_rs_strong_vassal_component add = var:tmp_rs_strong_vassal_step_neg }

        change_variable = { name = tmp_rs_succession_decay add = 19 }
        change_variable = { name = tmp_rs_succession_decay divide = 20 }
        if = { limit = { var:tmp_rs_succession_decay < 1 } set_variable = { name = tmp_rs_succession_decay value = 1 } }
        set_variable = { name = tmp_rs_succession_decay_neg value = var:tmp_rs_succession_decay }
        if = { limit = { var:riseandfall_rs_succession_shock > 0 } change_variable = { name = tmp_rs_succession_decay_neg multiply = -1 } }
        change_variable = { name = riseandfall_rs_succession_shock add = var:tmp_rs_succession_decay_neg }
        if = {
            limit = { var:riseandfall_rs_succession_shock = 0 }
            remove_variable = riseandfall_rs_succession_shock
        }

            clamp_variable = { name = riseandfall_rs_succession_shock min = -50 max = 50 }
        }

    riseandfall_apply_dynamic_story_pressure_se = yes

    if = { limit = { NOT = { has_variable = riseandfall_rs_legitimacy_component } } set_variable = { name = riseandfall_rs_legitimacy_component value = 0 } }
    if = { limit = { NOT = { has_variable = riseandfall_rs_legitimacy_multiplier } } set_variable = { name = riseandfall_rs_legitimacy_multiplier value = 100 } }
    if = { limit = { NOT = { has_variable = riseandfall_rs_vassals_component } } set_variable = { name = riseandfall_rs_vassals_component value = 0 } }
    if = { limit = { NOT = { has_variable = riseandfall_rs_popular_opinion_component } } set_variable = { name = riseandfall_rs_popular_opinion_component value = 0 } }
    if = { limit = { NOT = { has_variable = riseandfall_rs_realm_size_component } } set_variable = { name = riseandfall_rs_realm_size_component value = 0 } }
    if = { limit = { NOT = { has_variable = riseandfall_rs_strong_vassal_component } } set_variable = { name = riseandfall_rs_strong_vassal_component value = 0 } }
    if = { limit = { NOT = { has_variable = riseandfall_rs_culture_faith_component } } set_variable = { name = riseandfall_rs_culture_faith_component value = 0 } }
    if = { limit = { NOT = { has_variable = riseandfall_rs_epidemic_component } } set_variable = { name = riseandfall_rs_epidemic_component value = 0 } }
    if = { limit = { NOT = { has_variable = riseandfall_rs_partition_component } } set_variable = { name = riseandfall_rs_partition_component value = 0 } }
    if = { limit = { NOT = { has_variable = riseandfall_rs_succession_component } } set_variable = { name = riseandfall_rs_succession_component value = 0 } }
    if = { limit = { NOT = { has_variable = riseandfall_rs_story_pressure_component } } set_variable = { name = riseandfall_rs_story_pressure_component value = 0 } }
    if = { limit = { NOT = { has_variable = riseandfall_story_protection } } set_variable = { name = riseandfall_story_protection value = 0 } }
    if = { limit = { NOT = { has_variable = riseandfall_story_blob_pressure } } set_variable = { name = riseandfall_story_blob_pressure value = 0 } }
    # Start from 50 as the equilibrium midpoint; components shift toward 0 or 100
    set_variable = { name = tmp_target_score value = 50 }
    # Stats component contributes directly in range -30..+30 (already clamped above)
    set_variable = { name = tmp_stats_scaled value = var:riseandfall_rs_stats_component }
    change_variable = { name = tmp_target_score add = var:tmp_stats_scaled }
    # NOTE: legitimacy multiplier will be applied after realm law/vassal additions to scale the
    # final target (see below). For now we leave the multiplier application to the later stage.
    # Vassal relations are now 50% of calculation; add scaled vassal component mapped to -50..+50
    change_variable = { name = tmp_target_score add = var:riseandfall_rs_vassals_component }
    change_variable = { name = tmp_target_score add = var:riseandfall_rs_realm_size_component }
    change_variable = { name = tmp_target_score add = var:riseandfall_rs_strong_vassal_component }
    change_variable = { name = tmp_target_score add = var:riseandfall_rs_culture_faith_component }
    change_variable = { name = tmp_target_score add = var:riseandfall_rs_epidemic_component }
    change_variable = { name = tmp_target_score add = var:riseandfall_rs_partition_component }
    change_variable = { name = tmp_target_score add = var:riseandfall_rs_succession_component }
    change_variable = { name = tmp_target_score add = var:riseandfall_rs_story_pressure_component }
    if = {
        limit = { has_variable = riseandfall_story_successor_grace }
        change_variable = { name = tmp_target_score add = var:riseandfall_story_successor_grace }
    }
    # Story roles should protect realms through visible additive support only.
    # Do not apply hidden Anchor/Rival/Wildcard target floors here, because they make the tooltip math impossible to reconcile.
    # Successor grace is a visible flat bonus. Avoid a hidden target floor so the tooltip rows stay accountable.
    # Legitimacy is additive: -20 (none) to +15 (full). Transparent on the 0-100 scale.
    change_variable = { name = tmp_target_score add = var:riseandfall_rs_legitimacy_component }

    # Puppet stability bonus: -100..+100 while a puppet_regency is active.
    # Good puppet master stats replenish it; poor stats or exploitative actions drain it.
    if = {
        limit = { has_variable = riseandfall_rs_puppet_stability_bonus }
        change_variable = { name = tmp_target_score add = var:riseandfall_rs_puppet_stability_bonus }
    }

    # Civil war adjustment applied to the TARGET score (so smoothing moves current toward
    # the penalized target) instead of modifying the persistent current score directly.
    # Ensure counter exists
    if = { limit = { NOT = { has_variable = riseandfall_rs_civil_war_years } } set_variable = { name = riseandfall_rs_civil_war_years value = 0 } }
    # Ensure sex-law penalty tooltip var exists
    if = { limit = { NOT = { has_variable = riseandfall_realm_stability_sex_law_penalty } } set_variable = { name = riseandfall_realm_stability_sex_law_penalty value = 0 } }

    if = {
        limit = { is_independent_ruler = yes }
        # Dynamic civil war severity based on how much of the realm is in revolt.
        # We estimate revolt share as:
        #   sum(sub_realm_size of direct vassals at war with liege) / ruler realm_size
        # Tier mapping (percent rebel share):
        #   5-20 => +1 pressure/year, 21-40 => +2, 41-60 => +3, 61-80 => +4, 81-100 => +5
        # Pressure decays by 10%/year while not in civil war, with a minimum step of 1.
        # Final penalty is -pressure, capped at -50.
        save_scope_as = ruler
        set_variable = { name = tmp_cw_realm_counties value = 1 }
        set_variable = { name = tmp_cw_rebel_counties value = 0 }
        set_variable = { name = tmp_cw_rebel_percent value = 0 }
        set_variable = { name = tmp_cw_yearly_step value = 0 }

        if = {
            limit = { is_in_civil_war = yes }
            set_variable = { name = tmp_cw_realm_counties value = realm_size }
            if = { limit = { var:tmp_cw_realm_counties < 1 } set_variable = { name = tmp_cw_realm_counties value = 1 } }

            every_vassal = {
                limit = { is_at_war_with_liege = yes }
                save_scope_value_as = { name = tmp_cw_vassal_sub_realm value = sub_realm_size }
                scope:ruler = { change_variable = { name = tmp_cw_rebel_counties add = scope:tmp_cw_vassal_sub_realm } }
            }

            set_variable = { name = tmp_cw_rebel_percent value = var:tmp_cw_rebel_counties }
            change_variable = { name = tmp_cw_rebel_percent multiply = 100 }
            change_variable = { name = tmp_cw_rebel_percent divide = var:tmp_cw_realm_counties }
            clamp_variable = { name = tmp_cw_rebel_percent min = 0 max = 100 }

            if = { limit = { var:tmp_cw_rebel_percent >= 80 } set_variable = { name = tmp_cw_yearly_step value = 5 } }
            else_if = { limit = { var:tmp_cw_rebel_percent >= 61 } set_variable = { name = tmp_cw_yearly_step value = 4 } }
            else_if = { limit = { var:tmp_cw_rebel_percent >= 41 } set_variable = { name = tmp_cw_yearly_step value = 3 } }
            else_if = { limit = { var:tmp_cw_rebel_percent >= 21 } set_variable = { name = tmp_cw_yearly_step value = 2 } }

# Mark rulers right after succession and rate succession shock based on per-stat downgrades.
# Root scope: inheriting ruler
# Expected saved scope: scope:predecessor
riseandfall_mark_story_succession_shock_se = {
    if = {
        limit = {
            exists = scope:predecessor
        }
        # Predecessor may be alive in a stepdown/overthrow succession.
        set_variable = { name = tmp_rs_succession_shock value = 0 }

        # Each stat delta is capped at ±10.
        # We only want a penalty, so positive total deltas are stored as shock;
        # negative totals are clamped to 0 after aggregation.
        set_variable = { name = tmp_rs_stat_drop value = scope:predecessor.diplomacy }
        set_variable = { name = tmp_rs_current_stat value = diplomacy }
        change_variable = { name = tmp_rs_current_stat multiply = -1 }
        change_variable = { name = tmp_rs_stat_drop add = var:tmp_rs_current_stat }
        if = { limit = { var:tmp_rs_stat_drop > 10 } set_variable = { name = tmp_rs_stat_drop value = 10 } }
        else_if = { limit = { var:tmp_rs_stat_drop < -10 } set_variable = { name = tmp_rs_stat_drop value = -10 } }
        change_variable = { name = tmp_rs_succession_shock add = var:tmp_rs_stat_drop }

        set_variable = { name = tmp_rs_stat_drop value = scope:predecessor.martial }
        set_variable = { name = tmp_rs_current_stat value = martial }
        change_variable = { name = tmp_rs_current_stat multiply = -1 }
        change_variable = { name = tmp_rs_stat_drop add = var:tmp_rs_current_stat }
        if = { limit = { var:tmp_rs_stat_drop > 10 } set_variable = { name = tmp_rs_stat_drop value = 10 } }
        else_if = { limit = { var:tmp_rs_stat_drop < -10 } set_variable = { name = tmp_rs_stat_drop value = -10 } }
        change_variable = { name = tmp_rs_succession_shock add = var:tmp_rs_stat_drop }

        set_variable = { name = tmp_rs_stat_drop value = scope:predecessor.stewardship }
        set_variable = { name = tmp_rs_current_stat value = stewardship }
        change_variable = { name = tmp_rs_current_stat multiply = -1 }
        change_variable = { name = tmp_rs_stat_drop add = var:tmp_rs_current_stat }
        if = { limit = { var:tmp_rs_stat_drop > 10 } set_variable = { name = tmp_rs_stat_drop value = 10 } }
        else_if = { limit = { var:tmp_rs_stat_drop < -10 } set_variable = { name = tmp_rs_stat_drop value = -10 } }
        change_variable = { name = tmp_rs_succession_shock add = var:tmp_rs_stat_drop }

        set_variable = { name = tmp_rs_stat_drop value = scope:predecessor.intrigue }
        set_variable = { name = tmp_rs_current_stat value = intrigue }
        change_variable = { name = tmp_rs_current_stat multiply = -1 }
        change_variable = { name = tmp_rs_stat_drop add = var:tmp_rs_current_stat }
        if = { limit = { var:tmp_rs_stat_drop > 10 } set_variable = { name = tmp_rs_stat_drop value = 10 } }
        else_if = { limit = { var:tmp_rs_stat_drop < -10 } set_variable = { name = tmp_rs_stat_drop value = -10 } }
        change_variable = { name = tmp_rs_succession_shock add = var:tmp_rs_stat_drop }

        set_variable = { name = tmp_rs_stat_drop value = scope:predecessor.learning }
        set_variable = { name = tmp_rs_current_stat value = learning }
        change_variable = { name = tmp_rs_current_stat multiply = -1 }
        change_variable = { name = tmp_rs_stat_drop add = var:tmp_rs_current_stat }
        if = { limit = { var:tmp_rs_stat_drop > 10 } set_variable = { name = tmp_rs_stat_drop value = 10 } }
        else_if = { limit = { var:tmp_rs_stat_drop < -10 } set_variable = { name = tmp_rs_stat_drop value = -10 } }
        change_variable = { name = tmp_rs_succession_shock add = var:tmp_rs_stat_drop }

        clamp_variable = { name = tmp_rs_succession_shock min = -50 max = 50 }

        if = {
            limit = { var:tmp_rs_succession_shock > 0 }

# Inherit realm stability from predecessor
# Root scope: the heir/new character who should inherit
# scope:predecessor (or saved scope) should contain the character whose stability is being transferred
# This effect transfers 100% of the predecessor's stability score to the heir,
# then initializes the calculation for the new ruler (fresh components, then apply inherited bonus)
riseandfall_inherit_stability_score_se = {
    # Require that realm stability is enabled
    if = {
        limit = { has_game_rule = riseandfall_realm_stability_enabled }
        # Check if predecessor has a stability score to transfer
        if = {
            limit = {
                scope:predecessor = {
                    has_variable = riseandfall_realm_stability_score
                    var:riseandfall_realm_stability_score > 0
                }
            }
            # Save predecessor's score for calculation
            save_scope_value_as = {
                name = predecessor_stability_score
                value = scope:predecessor.var:riseandfall_realm_stability_score
            }
            # Save predecessor's score for tooltip
            save_scope_value_as = {
                name = predecessor_stability_amount
                value = scope:predecessor.var:riseandfall_realm_stability_score
            }
            
            # Transfer 100% of predecessor's stability to heir (initialize inherited score)
            # This becomes the starting point for the heir; it will be adjusted next recalculation
            set_variable = {
                name = riseandfall_realm_stability_score
                value = scope:predecessor_stability_score
            }
            # Also initialize other variables the heir might need
            set_variable = { name = riseandfall_rs_adjust_months value = 0 }
            # Copy companion history from predecessor: civil war years, puppet regency years, and war multiplier
            if = { limit = { scope:predecessor = { has_variable = riseandfall_rs_civil_war_years } } set_variable = { name = riseandfall_rs_civil_war_years value = scope:predecessor.var:riseandfall_rs_civil_war_years } }
            
            else = { set_variable = { name = riseandfall_rs_civil_war_years value = 0 } }
            if = { limit = { scope:predecessor = { has_variable = riseandfall_puppet_regency_years } } set_variable = { name = riseandfall_puppet_regency_years value = scope:predecessor.var:riseandfall_puppet_regency_years } }
            
            else = { set_variable = { name = riseandfall_puppet_regency_years value = 0 } }
            if = { limit = { scope:predecessor = { has_variable = riseandfall_realm_stability_war_multiplier } } set_variable = { name = riseandfall_realm_stability_war_multiplier value = scope:predecessor.var:riseandfall_realm_stability_war_multiplier } }
            
            else = { set_variable = { name = riseandfall_realm_stability_war_multiplier value = 100 } }
            if = { limit = { scope:predecessor = { has_variable = riseandfall_rs_difficulty_multiplier } } set_variable = { name = riseandfall_rs_difficulty_multiplier value = scope:predecessor.var:riseandfall_rs_difficulty_multiplier } }
            else = { set_variable = { name = riseandfall_rs_difficulty_multiplier value = 100 } }
            if = { limit = { scope:predecessor = { has_variable = riseandfall_rs_difficulty_multiplier_display } } set_variable = { name = riseandfall_rs_difficulty_multiplier_display value = scope:predecessor.var:riseandfall_rs_difficulty_multiplier_display } }
            else = { set_variable = { name = riseandfall_rs_difficulty_multiplier_display value = 1.0 } }
            # Legitimacy multiplier: new ruler starts fresh at 100%
            set_variable = { name = riseandfall_rs_legitimacy_multiplier value = 100 }
        }
        else = {
            # Predecessor had no stability score; initialize heir to baseline
            set_variable = { name = riseandfall_rs_adjust_months value = 0 }
            set_variable = { name = riseandfall_rs_civil_war_years value = 0 }
            set_variable = { name = riseandfall_puppet_regency_years value = 0 }
            set_variable = { name = riseandfall_realm_stability_war_multiplier value = 100 }
            set_variable = { name = riseandfall_rs_legitimacy_multiplier value = 100 }
        }
    }
}

# Merge an inherited realm's stored stability into an established ruler's score.
# Root scope: inheriting ruler with an existing stability score.
# Expected scopes: scope:predecessor has the inherited realm's score; scope:title is the gained realm title.
riseandfall_merge_inherited_realm_stability_se = {
    if = {
        limit = {
            has_game_rule = riseandfall_realm_stability_enabled
            has_variable = riseandfall_realm_stability_score
            exists = scope:predecessor
            scope:predecessor = { has_variable = riseandfall_realm_stability_score }
            exists = scope:title
            scope:title = {
                tier >= tier_duchy
                tier = scope:predecessor.highest_held_title_tier
            }
            realm_size >= 1
        }
        save_scope_as = rf_stability_inheritor
        set_variable = { name = tmp_rf_inherited_realm_counties value = 0 }
        if = {
            limit = {
                scope:predecessor = {
                    has_variable = riseandfall_rs_inheritance_realm_size
                    var:riseandfall_rs_inheritance_realm_size > 0
                }
            }
            set_variable = { name = tmp_rf_inherited_realm_counties value = scope:predecessor.var:riseandfall_rs_inheritance_realm_size }
        }
        else = {
            scope:title = {
                every_in_de_jure_hierarchy = {
                    limit = {
                        tier = tier_county
                        holder = {
                            OR = {
                                this = scope:rf_stability_inheritor
                                target_is_liege_or_above = scope:rf_stability_inheritor
                            }
                        }
                    }
                    scope:rf_stability_inheritor = {
                        change_variable = { name = tmp_rf_inherited_realm_counties add = 1 }
                    }
                }
            }
        }
        if = {
            limit = { var:tmp_rf_inherited_realm_counties > 0 }
            set_variable = { name = tmp_rf_total_realm_counties value = realm_size }
            if = {
                limit = { var:tmp_rf_inherited_realm_counties > var:tmp_rf_total_realm_counties }
                set_variable = { name = tmp_rf_inherited_realm_counties value = var:tmp_rf_total_realm_counties }
            }
            set_variable = { name = tmp_rf_existing_realm_counties value = var:tmp_rf_total_realm_counties }
            set_variable = { name = tmp_rf_inherited_realm_counties_neg value = var:tmp_rf_inherited_realm_counties }
            change_variable = { name = tmp_rf_inherited_realm_counties_neg multiply = -1 }
            change_variable = { name = tmp_rf_existing_realm_counties add = var:tmp_rf_inherited_realm_counties_neg }
            if = { limit = { var:tmp_rf_existing_realm_counties < 0 } set_variable = { name = tmp_rf_existing_realm_counties value = 0 } }

            set_variable = { name = tmp_rf_existing_stability_weighted value = var:riseandfall_realm_stability_score }
            change_variable = { name = tmp_rf_existing_stability_weighted multiply = var:tmp_rf_existing_realm_counties }
            set_variable = { name = tmp_rf_inherited_stability_weighted value = scope:predecessor.var:riseandfall_realm_stability_score }
            change_variable = { name = tmp_rf_inherited_stability_weighted multiply = var:tmp_rf_inherited_realm_counties }
            set_variable = { name = riseandfall_realm_stability_score value = var:tmp_rf_existing_stability_weighted }
            change_variable = { name = riseandfall_realm_stability_score add = var:tmp_rf_inherited_stability_weighted }
            change_variable = { name = riseandfall_realm_stability_score divide = var:tmp_rf_total_realm_counties }
            clamp_variable = { name = riseandfall_realm_stability_score min = 0 max = 100 }
            round_variable = { name = riseandfall_realm_stability_score nearest = 1 }

            set_variable = { name = riseandfall_rs_inherited_realm_share value = var:tmp_rf_inherited_realm_counties }
            change_variable = { name = riseandfall_rs_inherited_realm_share multiply = 100 }
            change_variable = { name = riseandfall_rs_inherited_realm_share divide = var:tmp_rf_total_realm_counties }
            clamp_variable = { name = riseandfall_rs_inherited_realm_share min = 0 max = 100 }
            round_variable = { name = riseandfall_rs_inherited_realm_share nearest = 1 }

            set_variable = { name = riseandfall_rs_inherited_realm_stability value = scope:predecessor.var:riseandfall_realm_stability_score }
            clamp_variable = { name = riseandfall_rs_inherited_realm_stability min = 0 max = 100 }
            round_variable = { name = riseandfall_rs_inherited_realm_stability nearest = 1 }

            # The immediate merge is the inheritance adjustment. Let the following
            # recalculation refresh components and modifiers without applying an extra smoothing step.
            set_variable = { name = riseandfall_rs_adjust_months value = -1 }
            set_variable = { name = riseandfall_realm_stability_predicted_change value = 0 }
            set_variable = { name = riseandfall_rs_last_stability_merge_predecessor value = scope:predecessor }
        }
    }
}

# Called on the war winner scope (character) to reduce the war penalty by -10
riseandfall_war_victory_se = {
    # Defensive: only operate on living characters
    if = { limit = { is_alive = yes }
    if = { limit = { NOT = { has_variable = riseandfall_war_penalty } } set_variable = { name = riseandfall_war_penalty value = 0 } }
    # Defensive: if a war scope exists and it's a peasant war, don't apply victory penalty reduction.
    if = {
        limit = { scope:war = { using_cb = peasant_war } }
        # do nothing for peasant wars
    }
    else = {
        # Reduce penalty by -10
        change_variable = { name = riseandfall_war_penalty add = -10 }
        # Clamp to minimum 0
        if = { limit = { var:riseandfall_war_penalty < 0 } set_variable = { name = riseandfall_war_penalty value = 0 } }
    }
    }
}

        limit = {
            is_alive = yes
            is_landed = yes
            OR = {
                has_trait = peasant_leader
                has_trait = populist_leader
            }
        }
        every_realm_county = {
            riseandfall_clear_low_stability_county_modifiers_se = yes
        }
    }
}

# Called on the war loser scope (character) to increase the war penalty by +10
riseandfall_war_loss_se = {
    # Only operate on living characters; variables are stored on character scopes
    if = {
        limit = { is_alive = yes }
        if = { limit = { NOT = { has_variable = riseandfall_war_penalty } } set_variable = { name = riseandfall_war_penalty value = 0 } }
        # Defensive: if a war scope exists and it's a peasant war, don't apply loss penalty here
        if = {
            limit = { scope:war = { using_cb = peasant_war } }
            # do nothing for peasant wars
        }
        else = {
            # Increase penalty by +10
            change_variable = { name = riseandfall_war_penalty add = 10 }
        }
    }
}
