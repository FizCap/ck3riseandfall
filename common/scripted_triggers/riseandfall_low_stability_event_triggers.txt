# Rise and Fall - Low Stability Event Availability Triggers
# Keep per-event eligibility logic here so the selector stays simple.

riseandfall_low_stability_warlord_available_trigger = {
    NOT = { exists = global_var:riseandfall_low_stability_event_warlord_disabled }
    # must have at least one landed vassals of county tier or higher
    any_vassal = {
        count >= 1
        is_landed = yes
        primary_title.tier >= tier_county
    }
}

riseandfall_low_stability_palace_coup_available_trigger = {
    NOT = { exists = global_var:riseandfall_low_stability_event_palace_coup_disabled }
    NOT = { government_has_flag = government_is_theocracy }
    NOT = { government_has_flag = government_is_republic }
    NOT = { has_diarchy_type = puppet_regency }
}

riseandfall_low_stability_realm_split_available_trigger = {
    riseandfall_can_realm_split_trigger = yes
    NOT = { exists = global_var:riseandfall_low_stability_event_realm_split_disabled }
}

riseandfall_low_stability_authority_collapse_available_trigger = {
    NOT = { exists = global_var:riseandfall_low_stability_event_authority_collapse_disabled }
    NOT = {
        any_realm_county = {
            has_county_modifier = riseandfall_authority_collapse
        }
    }
}
