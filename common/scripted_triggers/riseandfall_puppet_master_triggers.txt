# Liege stability below 30 (recipient variable)
riseandfall_liege_stability_below_30 = {
	# Actual condition: check recipient's stored stability score
	scope:recipient = {
		# Guard against unset variable to avoid engine errors when stability hasn't been initialized
		has_variable = riseandfall_realm_stability_score
		var:riseandfall_realm_stability_score < 30
	}
	# Show a friendly tooltip when this trigger fails
	custom_tooltip = {
		text = riseandfall_liege_stability_below_30
		scope:recipient = {
			has_variable = riseandfall_realm_stability_score
			var:riseandfall_realm_stability_score < 30
		}
	}
}

# No cooldown active
riseandfall_no_cooldown = {
	# Actual condition: no puppet master cooldown variable present on the recipient
	scope:recipient = { NOT = { has_variable = puppet_master_cooldown } }
	# Friendly failure text for cooldown checks
	custom_tooltip = {
		text = riseandfall_no_cooldown
		scope:recipient = { NOT = { has_variable = puppet_master_cooldown } }
	}
}
