# Quarterly fallback: re-sync puppet master court position for any liege whose position
# is missing while a puppet_regency is still active. Ensures refill within 3 months
# if the 1-day revoke event somehow fails.
quarterly_playable_pulse = {
    on_actions = {
        riseandfall_puppet_master_quarterly_resync
    }
}

riseandfall_puppet_master_quarterly_resync = {
    trigger = {
        has_game_rule = riseandfall_low_stability_events_enabled
        has_diarchy_type = puppet_regency
        has_active_diarchy = yes
        NOT = { employs_court_position = riseandfall_puppet_master_court_position }
    }
    effect = {
        riseandfall_sync_puppet_master_court_position = yes
    }
}

riseandfall_low_stability_roll = {
    trigger = {
        has_game_rule = riseandfall_low_stability_events_enabled
    }
    effect = {
        every_independent_ruler = {
            limit = {
                is_landed = yes
                # Only rulers whose primary title is duchy or higher (duke+)
                primary_title.tier >= tier_duchy
            }
            save_scope_as = ruler_low_stability
            riseandfall_low_stability_roll_se = yes
        }
    }
}

    trigger = {
        has_game_rule = riseandfall_low_stability_events_enabled
    }
    effect = {
        riseandfall_check_puppet_regency_end = yes
    }
}


riseandfall_warlord = {
    trigger = {
        has_game_rule = riseandfall_low_stability_events_enabled
    }
    effect = {
        # Cleanup: handle unlanded/independent warlords
        riseandfall_warlord_cleanup_se = yes

        # Yearly reconsideration: fire warlord.1003 once per year for each AI warlord vassal
        every_ruler = {
            limit = {
                is_ai = yes
                is_independent_ruler = no
                is_landed = yes
                has_trait = warlord
                NOT = { has_variable = riseandfall_warlord_reconsider_cooldown }
            }
            # Set cooldown BEFORE firing so rapid re-calls cannot double-queue
            set_variable = { name = riseandfall_warlord_reconsider_cooldown value = yes years = 1 }
            trigger_event = { id = warlord.1003 }
        }
    }
}

# Rise and Fall: respond to imprison actions
# root = prisoner, scope:imprisoner = the character who imprisoned them
# If a warlord is imprisoned by their liege, remove the warlord trait so they stop behaving as an independent warlord
# Call the scripted effect which handles warlord trait removal when a liege imprisons a vassal
riseandfall_warlord_imprisoned = {
    effect = {
        # root = prisoner, scope:imprisoner = the character who imprisoned them
        # Call the scripted effect which handles warlord trait removal when a liege imprisons a vassal
        riseandfall_warlord_imprisoned_se = yes
    }
}


# On_actions for puppet masters (diarchs): AI-only unlanded diarchs should be given spouses by their liege's court.
# We iterate each ruler with a puppet_regency diarchy and pick any courtier who is that ruler's diarch, then call the marriage scripted effects.
riseandfall_marry_male_puppet_master_target = {
    effect = {
        every_ruler = {
            limit = { has_diarchy_type = puppet_regency }
            save_scope_as = liege
            every_courtier = {
                limit = {
                    is_diarch_of_target = scope:liege
                    is_ai = yes
                    is_landed = no
                    is_ruler = no
                    NOT = { has_trait_with_flag = can_not_marry }
                    is_male = yes
                    is_adult = yes
                    is_married = no
                    is_betrothed = no
                }
                save_scope_as = diarch_candidate
                # call the scripted effect on the diarch character (root)
                riseandfall_marry_puppet_master_male_se = yes
            }
        }
    }
}

riseandfall_split_title_monitor = {
    trigger = {
        has_game_rule = riseandfall_low_stability_events_enabled
    }
    effect = {
        every_ruler = {
            limit = {
                is_landed = yes
                exists = primary_title
            }
            save_scope_as = rf_ruler
            scope:rf_ruler = {
                riseandfall_split_title_monitor_cleanup_se = yes
            }
        }
    }
}
