yearly_global_pulse = {
    on_actions = {
        riseandfall_admin_score
        riseandfall_admin_score_cleanup
    }
}

on_game_start_after_lobby = {
    on_actions = {
        riseandfall_admin_score
        riseandfall_admin_score_cleanup
    }
}

# On_Actions

riseandfall_admin_score = {
    # Only run the admin random score if the game rule is enabled
    trigger = { has_game_rule = riseandfall_admin_score_nerf_enabled }
    effect = {
        riseandfall_random_score = yes
    }
}

riseandfall_admin_score_cleanup = {
    # If the rule is disabled, clear stale admin score variables from all characters
    trigger = { has_game_rule = riseandfall_admin_score_nerf_disabled }
    effect = {
        riseandfall_clear_admin_nerf = yes
    }
}
